105 static int register_if_ddf_not_ready(ddt_t * ddt, ddf_t * ddf) {
106  int registered = 0;
107  ddt_t * headDDTWaitList = (ddt_t *) ddf->headDDTWaitList;
108  while (headDDTWaitList != DDF_SATISFIED && !registered) {
109  /* headDDTWaitList can not be DDF_SATISFIED in here */
110  ddt->nextDDTWaitingOnSameDDF = headDDTWaitList;
111 
112  // Try to add the ddt to the ddf's list of ddts waiting.
113  registered = __sync_bool_compare_and_swap(&(ddf->headDDTWaitList),
114  headDDTWaitList, ddt);
115 
116  /* may have failed because either some other task tried to be the head or a put occurred. */
117  if (!registered) {
118  headDDTWaitList = (ddt_t *) ddf->headDDTWaitList;
119  /* if waitListOfDDF was set to DDF_SATISFIED, the loop condition will handle that
120  * if another task was added, now try to add in front of that
121  */
122  }
123  }
124  if (DEBUG_DDF && registered) {
125  printf("ddf: %p registered %p as headDDT of ddf %p\n", ddt, ddf->headDDTWaitList, ddf);
126  }
127  return registered;
128 }

134 int iterate_ddt_frontier(ddt_t * ddt) {
135  ddf_t ** currentDDF = ddt->waitingFrontier;
136 
137  if (DEBUG_DDF) {
138  printf("ddf: Trying to iterate ddt %p, frontier ddf is %p \n", ddt, currentDDF);
139  }
140  // Try and register on ddf until one is not ready.
141  while ((*currentDDF) && !register_if_ddf_not_ready(ddt, *currentDDF)) {
142  ++currentDDF;
143  }
144  // Here we either:
145  // - Have hit a ddf not ready (i.e. no 'put' on this one yet)
146  // - iterated over the whole list (i.e. all 'put' must have happened)
147  if (DEBUG_DDF) { printf("ddf: iterate result %d\n", (*currentDDF == NULL)); }
148  ddt->waitingFrontier = currentDDF;
149  return (*currentDDF == NULL);
150 }

168 void ddf_put(ddf_t * ddf, void * datum) {
169  assert(datum != UNINITIALIZED_DDF_DATA_PTR && EMPTY_DATUM_ERROR_MSG);
170  assert(ddf != NULL && "error: can't DDF is a pointer to NULL");
171  //TODO Limitation: not enough to guarantee single assignment
172  if (DEBUG_DDF) { printf("ddf: put datum %p\n", ddf->datum); }
173  assert(ddf->datum == NULL && "error: violated single assignment property for DDFs");
174 
175  volatile ddt_t * headDDTWaitList = NULL;
176  ddf->datum = datum;
177  headDDTWaitList = ddf->headDDTWaitList;
178  if (DEBUG_DDF) { printf("ddf: Retrieve ddf %p head's ddt %p\n", ddf, headDDTWaitList); }
179  // Try to cas the wait list to prevent new DDTs from registering on this ddf.
180  // When cas successed, new DDTs see the DDF has been satisfied and proceed.
181  while (!__sync_bool_compare_and_swap(&(ddf->headDDTWaitList),
182  (struct ddt_st *) headDDTWaitList, DDF_SATISFIED )) {
183  headDDTWaitList = ddf->headDDTWaitList;
184  }
185  if (DEBUG_DDF) {
186  printf("ddf: Put successful on ddf %p set head's ddf to %p\n", ddf, ddf->headDDTWaitList);
187  }
188  // Now iterate over the list of DDTs, and try to advance their frontier.
189  ddt_t * currDDT = (ddt_t *) headDDTWaitList;
190  while (currDDT != UNINITIALIZED_DDF_WAITLIST_PTR) {
191  if (DEBUG_DDF) { printf("ddf: Trying to iterate frontier of ddt %p\n", currDDT); }
192  if (iterate_ddt_frontier(currDDT)) {
193  // DDT eligible to scheduling
194  async_task_t * async_task = rt_ddt_to_async_task(currDDT);
195  if (DEBUG_DDF) { printf("ddf: async_task %p\n", async_task); }
196  try_schedule_async(async_task);
197  }
198  currDDT = currDDT->nextDDTWaitingOnSameDDF;
199  }
