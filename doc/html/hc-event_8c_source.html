80 // Generic initializer for events. Implementation-dependent event functions
81 // are passed through the function pointer data-structure.
82 static ocrEvent_t* eventConstructorInternal(ocrPolicyDomain_t * pd, ocrEventFactory_t * factory, ocrEventTypes_t eventType, bool takesArg) {
83  ocrEvent_t* base = NULL;
84  ocrEventFcts_t * eventFctPtrs = NULL;
85  if (eventType == OCR_EVENT_FINISH_LATCH_T) {
86  ocrEventHcFinishLatch_t * eventImpl = (ocrEventHcFinishLatch_t*) checkedMalloc(eventImpl, sizeof(ocrEventHcFinishLatch_t));
87  eventImpl->counter = 0;
88  //Note: waiters are initialized afterwards
89  eventFctPtrs = &(((ocrEventFactoryHc_t*)factory)->finishLatchFcts);
90  base = (ocrEvent_t*)eventImpl;
91  } else if (eventType == OCR_EVENT_LATCH_T) {
92  ocrEventHcLatch_t * eventImpl = (ocrEventHcLatch_t*) checkedMalloc(eventImpl, sizeof(ocrEventHcLatch_t));
93  (eventImpl->base).waiters = END_OF_LIST;
94  (eventImpl->base).signalers = END_OF_LIST;
95  (eventImpl->base).data = NULL_GUID;
96  eventImpl->counter = 0;
97  eventFctPtrs = &(factory->latchFcts);
98  base = (ocrEvent_t*)eventImpl;
99  } else {
100  ASSERT(((eventType == OCR_EVENT_ONCE_T) ||
101  (eventType == OCR_EVENT_IDEM_T) ||
102  (eventType == OCR_EVENT_STICKY_T)) &&
103  "error: Unsupported type of event");
104  ocrEventHcSingle_t* eventImpl;
105  if (eventType == OCR_EVENT_ONCE_T) {
106  ocrEventHcOnce_t* onceImpl = (ocrEventHcOnce_t*) checkedMalloc(eventImpl, sizeof(ocrEventHcOnce_t));
107  onceImpl->nbEdtRegistered = pd->getAtomic64(pd, NULL);
108  eventImpl = (ocrEventHcSingle_t*) onceImpl;
109  } else {
110  eventImpl = (ocrEventHcSingle_t*) checkedMalloc(eventImpl, sizeof(ocrEventHcSingle_t));
111  }
112  (eventImpl->base).waiters = END_OF_LIST;
113  (eventImpl->base).signalers = END_OF_LIST;
114  (eventImpl->base).data = UNINITIALIZED_DATA;
115  eventFctPtrs = &(factory->singleFcts);
116  base = (ocrEvent_t*)eventImpl;
117  }
118 
119  // Initialize ocrEventHc_t
120  base->kind = eventType;
121 
122  // Initialize ocrEvent_t base
123  base->fctPtrs = eventFctPtrs;
124  base->guid = UNINITIALIZED_GUID;
125  guidify(pd, (u64)base, &(base->guid), OCR_GUID_EVENT);
126  DPRINTF(DEBUG_LVL_INFO, "Create %s: 0x%lx\n", eventTypeToString(base), base->guid);
127  return base;
128 }

130 static void destructEventHc ( ocrEvent_t* base ) {
131  // Event's signaler/waiter must have been previously deallocated
132  // at some point before. For instance on satisfy.
133  DPRINTF(DEBUG_LVL_INFO, "Destroy %s: 0x%lx\n", eventTypeToString(base), base->guid);
134  ocrEventHc_t* derived = (ocrEventHc_t*)base;
135  ocrPolicyDomain_t *pd = getCurrentPD();
136  ocrPolicyCtx_t * orgCtx = getCurrentWorkerContext();
137  ocrPolicyCtx_t * ctx = orgCtx->clone(orgCtx);
138  ctx->type = PD_MSG_GUID_REL;
139  pd->inform(pd, base->guid, ctx);
140  if(base->kind == OCR_EVENT_ONCE_T) {
141  ocrEventHcOnce_t * onceEvent = (ocrEventHcOnce_t *) base;
142  onceEvent->nbEdtRegistered->fctPtrs->destruct(onceEvent->nbEdtRegistered);
143  }
144  ctx->destruct(ctx);
145  free(derived);
146 }

153 // Set a single event data guid
154 // Warning: Concurrent with registration on sticky event.
155 static regNode_t * singleEventPut(ocrEventHcAwaitable_t * self, ocrGuid_t data ) {
156  // TODO This is not enough to always detect concurrent puts.
157  ASSERT (self->data == UNINITIALIZED_DATA && "violated single assignment property for EDFs");
158  self->data = data;
159  // list of entities waiting on the event
160  volatile regNode_t* waiters = self->waiters;
161  // 'put' may be concurrent to another entity trying to register to this event
162  // Try to cas the waiters list to SEALED_LIST to let other know the 'put' has occurred.
163  while ( !__sync_bool_compare_and_swap( &(self->waiters), waiters, SEALED_LIST)) {
164  // If we failed to cas, it means a new node has been inserted
165  // in front of the registration list, so update our pointer and try again.
166  waiters = self->waiters;
167  }
168  // return the most up to date head on the registration list
169  // (no more adds possible once SEALED_LIST has been set)
170  ASSERT(self->waiters == SEALED_LIST);
171  return (regNode_t *) waiters;
172 }

174 // This is setting a sticky event's data
175 // slotEvent is ignored for stickies.
176 static void singleEventSatisfy(ocrEvent_t * base, ocrGuid_t data, u32 slotEvent) {
177  ocrEventHcAwaitable_t * self = (ocrEventHcAwaitable_t *) base;
178 
179  // Whether it is a once, idem or sticky, unitialized means it's the first
180  // time we try to satisfy the event. Note: It's a very loose check, the
181  // 'Put' implementation must do more work to detect races on data.
182  if (self->data == UNINITIALIZED_DATA) {
183  DPRINTF(DEBUG_LVL_INFO, "Satisfy %s: 0x%lx with 0x%lx\n", eventTypeToString(base), base->guid, data);
184  // Single events don't have slots, just put the data
185  regNode_t * waiters = singleEventPut(self, data);
186  // Put must have sealed the waiters list and returned it
187  // Note: Here the design is not great, 'put' CAS and returns the
188  // waiters list but does not handle the signaler list which
189  // is ok for now because it's not used.
190  ASSERT(waiters != SEALED_LIST);
191  ASSERT(self->data != UNINITIALIZED_DATA);
192  // Need to signal other entities waiting on the event
193  regNode_t * waiter = waiters;
194  while (waiter != END_OF_LIST) {
195  // Note: in general it's better to read from self->data in case
196  // the event does some post processing on the input data
197  signalWaiter(waiter->guid, self->data, waiter->slot);
198  waiters = waiter;
199  waiter = waiter->next;
200  free(waiters); // Release waiter node
201  }
202  } else {
203  // once-events cannot survive down here
204  // idem-events ignore extra satisfy
205  if (base->kind == OCR_EVENT_STICKY_T) {
206  ASSERT(false && "error: Multiple satisfy on a sticky event");
207  }
208  }
209 }

222 // Runs concurrently with registration on latch
223 static void latchEventSatisfy(ocrEvent_t * base, ocrGuid_t data, u32 slot) {
224  ASSERT((slot == OCR_EVENT_LATCH_DECR_SLOT) || (slot == OCR_EVENT_LATCH_INCR_SLOT));
225  ocrEventHcLatch_t * latch = (ocrEventHcLatch_t *) base;
226  u32 incr = (slot == OCR_EVENT_LATCH_DECR_SLOT) ? -1 : 1;
227  u32 count;
228  do {
229  count = latch->counter;
230  } while(!__sync_bool_compare_and_swap(&(latch->counter), count, count+incr));
231 
232  DPRINTF(DEBUG_LVL_INFO, "Satisfy %s: 0x%lx %s\n", eventTypeToString(base), base->guid, ((slot == OCR_EVENT_LATCH_DECR_SLOT) ? "decr":"incr"));
233  if ((count+incr) == 0) {
234  DPRINTF(DEBUG_LVL_INFO, "Satisfy %s: 0x%lx reached zero\n", eventTypeToString(base), base->guid);
235  ocrEventHcAwaitable_t * self = (ocrEventHcAwaitable_t *) base;
236  //TODO add API to seal a list
237  //TODO: do we need volatile here ?
238  // CAS the wait-list
239  volatile regNode_t* waiters = self->waiters;
240  // satisfy is concurrent with registration on the event
241  // Try to cas the waiters list to SEALED_LIST to let others know the
242  // latch has been satisfied
243  while ( !__sync_bool_compare_and_swap( &(self->waiters), waiters, SEALED_LIST)) {
244  // If we failed to cas, it means a new node has been inserted
245  // in front of the registration list, so update our pointer and try again.
246  waiters = self->waiters;
247  }
248  ASSERT(self->waiters == SEALED_LIST);
249  regNode_t * waiter = (regNode_t*) waiters;
250  // Go over the list and notify waiters
251  while (waiter != END_OF_LIST) {
252  // For now latch events don't have any data output
253  signalWaiter(waiter->guid, NULL_GUID, waiter->slot);
254  waiters = waiter;
255  waiter = waiter->next;
256  free((regNode_t*) waiters); // Release waiter node
257  }
258  }
259 }

267 //
268 // OCR-HC Finish-Latch Events Implementation
269 //
270 
271 // Requirements:
272 // R1) All dependences (what the finish latch will satisfy) are provided at creation. This implementation DOES NOT support outstanding registrations.
273 // R2) Number of incr and decr signaled on the event MUST BE equal.
274 static void finishLatchEventSatisfy(ocrEvent_t * base, ocrGuid_t data, u32 slot) {
275  ASSERT((slot == OCR_EVENT_LATCH_DECR_SLOT) || (slot == OCR_EVENT_LATCH_INCR_SLOT));
276  ocrEventHcFinishLatch_t * self = (ocrEventHcFinishLatch_t *) base;
277  u32 incr = (slot == OCR_EVENT_LATCH_DECR_SLOT) ? -1 : 1;
278  u32 count;
279  do {
280  count = self->counter;
281  } while(!__sync_bool_compare_and_swap(&(self->counter), count, count+incr));
282  DPRINTF(DEBUG_LVL_VERB, "Satisfy %s: 0x%lx %s\n", eventTypeToString(base), base->guid, ((slot == OCR_EVENT_LATCH_DECR_SLOT)? "decr":"incr"));
283  // No possible race when we reached 0 (see R2)
284  if ((count+incr) == 0) {
285  DPRINTF(DEBUG_LVL_INFO, "Satisfy %s: 0x%lx reached zero\n", eventTypeToString(base), base->guid);
286  // Important to void the ELS at that point, to make sure there's no
287  // side effect on code executing downwards.
288  ocrTask_t * task = getCurrentTask();
289  setFinishLatch(task, NULL_GUID);
290  // Notify waiters the latch is satisfied (We can extend that to a list // of waiters if we need to. (see R1))
291  // Notify output event if any associated with the finish-edt
292  regNode_t * outputEventWaiter = &(self->outputEventWaiter);
293  if (outputEventWaiter->guid != NULL_GUID) {
294  signalWaiter(outputEventWaiter->guid, self->returnGuid, outputEventWaiter->slot);
295  }
296  // Notify the parent latch if any
297  regNode_t * parentLatchWaiter = &(self->parentLatchWaiter);
298  if (parentLatchWaiter->guid != NULL_GUID) {
299  ocrEvent_t * parentLatch;
300  deguidify(getCurrentPD(), parentLatchWaiter->guid, (u64*)&parentLatch, NULL);
301  finishLatchCheckout(parentLatch);
302  }
303  // Since finish-latch is internal to finish-edt, and ELS is cleared,
304  // there are no more pointers left to it, deallocate.
305  base->fctPtrs->destruct(base);
306  }
307 }
