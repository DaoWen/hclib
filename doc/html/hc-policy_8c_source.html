45 static void hcPolicyDomainStart(ocrPolicyDomain_t * policy) {
46  // The PD should have been brought up by now and everything instantiated
47  // WARNING: Threads start should be the last thing we do here after
48  // all data-structures have been initialized.
49  u64 i = 0;
50  u64 workerCount = policy->workerCount;
51  u64 computeCount = policy->computeCount;
52  u64 schedulerCount = policy->schedulerCount;
53  ASSERT(workerCount == computeCount);
54 
55  // Start the allocators
56  for(i = 0; i < policy->allocatorCount; ++i) {
57  policy->allocators[i]->fctPtrs->start(policy->allocators[i], policy);
58  }
59 
60  // Start schedulers
61  for(i = 0; i < schedulerCount; i++) {
62  policy->schedulers[i]->fctPtrs->start(policy->schedulers[i], policy);
63  }
64 
65  //TODO workers could be responsible for starting the underlying target
66  // Note: it's important to first logically start all workers.
67  // Once they are all up, start the runtime
68  // Only start (N-1) workers as worker '0' is the current thread.
69  for(i = 1; i < workerCount; i++) {
70  policy->workers[i]->fctPtrs->start(policy->workers[i], policy);
71  }
72 
73  // Need to associate thread and worker here, as current thread fall-through
74  // in user code and may need to know which Worker it is associated to.
75  // Handle target '0'
76  policy->workers[0]->fctPtrs->start(policy->workers[0], policy);
77 }

90 static void hcPolicyDomainStop(ocrPolicyDomain_t * policy) {
91  // In HC, we MUST call stop on the master worker first.
92  // The master worker enters its work routine loop and will
93  // be unlocked by ocrShutdown
94 
95  policy->workers[0]->fctPtrs->stop(policy->workers[0]);
96  // WARNING: Do not add code here unless you know what you're doing !!
97  // If we are here, it means an EDT called ocrShutdown which
98  // logically finished workers and can make thread '0' executes this
99  // code before joining the other threads.
100 
101  // Thread '0' joins the other (N-1) threads.
102 
103  u64 i;
104  for (i = 1; i < policy->workerCount; i++) {
105  policy->workers[i]->fctPtrs->stop(policy->workers[i]);
106  }
107 }

109 static void hcPolicyDomainDestruct(ocrPolicyDomain_t * policy) {
110  // Destroying instances
111  int i = 0;
112  ocrScheduler_t ** schedulers = policy->schedulers;
113  for ( i = 0; i < policy->schedulerCount; ++i ) {
114  schedulers[i]->fctPtrs->destruct(schedulers[i]);
115  }
116  ocrWorker_t ** workers = policy->workers;
117  for ( i = 0; i < policy->workerCount; ++i ) {
118  workers[i]->fctPtrs->destruct(workers[i]);
119  }
120  ocrCompTarget_t ** computes = policy->computes;
121  for ( i = 0; i < policy->computeCount; ++i ) {
122  computes[i]->fctPtrs->destruct(computes[i]);
123  }
124  ocrWorkpile_t ** workpiles = policy->workpiles;
125  for ( i = 0; i < policy->workpileCount; ++i ) {
126  workpiles[i]->fctPtrs->destruct(workpiles[i]);
127  }
128  ocrAllocator_t ** allocators = policy->allocators;
129  for ( i = 0; i < policy->allocatorCount; ++i ) {
130  allocators[i]->fctPtrs->destruct(allocators[i]);
131  }
132  ocrMemTarget_t ** memories = policy->memories;
133  for ( i = 0; i < policy->memoryCount; ++i ) {
134  memories[i]->fctPtrs->destruct(memories[i]);
135  }
136 
137  // Simple hc policies don't have neighbors
138  ASSERT(policy->neighbors == NULL);
139 
140  // Destruct factories after instances as the instances
141  // rely on a structure in the factory (the function pointers)
142  policy->taskFactory->destruct(policy->taskFactory);
143  policy->taskTemplateFactory->destruct(policy->taskTemplateFactory);
144  policy->dbFactory->destruct(policy->dbFactory);
145  policy->eventFactory->destruct(policy->eventFactory);
146  policy->lockFactory->destruct(policy->lockFactory);
147  policy->atomicFactory->destruct(policy->atomicFactory);
148 
149  //Anticipate those to be null-impl for some time
150  ASSERT(policy->costFunction == NULL);
151 
152  // Finish with those in case destruct implementation needs
153  // to releaseGuids or access context for some reasons
154  policy->contextFactory->destruct(policy->contextFactory);
155  policy->guidProvider->fctPtrs->destruct(policy->guidProvider);
156 
157  free(policy);
158 }

160 static u8 hcAllocateDb(ocrPolicyDomain_t *self, ocrGuid_t *guid, void** ptr, u64 size,
161  u16 properties, ocrGuid_t affinity, ocrInDbAllocator_t allocator,
162  ocrPolicyCtx_t *context) {
163 
164  // Currently a very simple model of just going through all allocators
165  u64 i;
166  void* result;
167  for(i=0; i < self->allocatorCount; ++i) {
168  result = self->allocators[i]->fctPtrs->allocate(self->allocators[i],
169  size);
170  if(result) break;
171  }
172  // TODO: return error code. Requires our own errno to be clean
173  if(i < self->allocatorCount) {
174  ocrDataBlock_t *block = self->dbFactory->instantiate(self->dbFactory,
175  self->allocators[i]->guid, self->guid,
176  size, result, properties, NULL);
177  *ptr = result;
178  *guid = block->guid;
179  return 0;
180  }
181  return 1; // TODO: Return ENOMEM
182 }
