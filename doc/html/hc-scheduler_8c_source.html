37 static void hcSchedulerStart(ocrScheduler_t * self, ocrPolicyDomain_t * PD) {
38  ocrSchedulerHc_t * derived = (ocrSchedulerHc_t *) self;
39  u64 workpileCount = self->workpileCount;
40  ocrWorkpile_t ** workpiles = self->workpiles;
41  // allocate steal iterator cache
42  ocrWorkpileIterator_t ** stealIteratorsCache = checkedMalloc(stealIteratorsCache, sizeof(ocrWorkpileIterator_t *)*workpileCount);
43  // Initialize steal iterator cache
44  u64 i = 0;
45  while(i < workpileCount) {
46  // Note: here we assume workpile 'i' will match worker 'i' => Not great
47  stealIteratorsCache[i] = newWorkpileIterator(i, workpileCount, workpiles);
48  i++;
49  }
50  derived->stealIterators = stealIteratorsCache;
51 }

57 static u8 hcSchedulerYield (ocrScheduler_t* self, ocrGuid_t workerGuid,
58  ocrGuid_t yieldingEdtGuid, ocrGuid_t eventToYieldForGuid,
59  ocrGuid_t * returnGuid,
60  ocrPolicyCtx_t *context) {
61  // We do not yet take advantage of knowing which EDT we are yielding for.
62  ocrPolicyDomain_t * pd = context->PD;
63  ocrWorker_t * worker = NULL;
64  deguidify(pd, workerGuid, (u64*)&(worker), NULL);
65  // Retrieve currently executing edt's guid
66  ocrEvent_t * eventToYieldFor = NULL;
67  deguidify(pd, eventToYieldForGuid, (u64*)&(eventToYieldFor), NULL);
68 
69  ocrPolicyCtx_t * orgCtx = getCurrentWorkerContext();
70  ocrPolicyCtx_t * ctx = orgCtx->clone(orgCtx);
71  ctx->type = PD_MSG_EDT_TAKE;
72 
73  ocrGuid_t result = ERROR_GUID;
74  //This only works for single events, not latches
75  ASSERT(isEventSingleGuid(eventToYieldForGuid));
76  while((result = eventToYieldFor->fctPtrs->get(eventToYieldFor, 0)) == ERROR_GUID) {
77  u32 count;
78  ocrGuid_t taskGuid;
79  pd->takeEdt(pd, NULL, &count, &taskGuid, ctx);
80  ASSERT(count <= 1); // >1 not yet supported
81  if (count != 0) {
82  ocrTask_t* task = NULL;
83  deguidify(pd, taskGuid, (u64*)&(task), NULL);
84  worker->fctPtrs->execute(worker, task, taskGuid, yieldingEdtGuid);
85  }
86  }
87  *returnGuid = result;
88  ctx->destruct(ctx);
89  return 0;
90 }

92 static u8 hcSchedulerTake (ocrScheduler_t *self, struct _ocrCost_t *cost, u32 *count,
93  ocrGuid_t *edts, ocrPolicyCtx_t *context) {
94  // In this implementation (getCurrentPD == context->sourcePD)
95  // Source must be a worker guid and we rely on indices to map
96  // workers to workpiles (one-to-one)
97  u64 workerId = context->sourceId;
98  // First try to pop
99  ocrWorkpile_t * wp_to_pop = popMappingOneToOne(self, workerId);
100  // TODO sagnak, just to get it to compile, I am trickling down the 'cost' though it most probably is not the same
101  ocrGuid_t popped = wp_to_pop->fctPtrs->pop(wp_to_pop,cost);
102  if ( NULL_GUID == popped ) {
103  // If popping failed, try to steal
104  ocrWorkpileIterator_t* it = stealMappingOneToAllButSelf(self, workerId);
105  while ( it->hasNext(it) && (NULL_GUID == popped)) {
106  ocrWorkpile_t * next = it->next(it);
107  // TODO sagnak, just to get it to compile, I am trickling down the 'cost' though it most probably is not the same
108  popped = next->fctPtrs->steal(next, cost);
109  }
110  // Note that we do not need to destruct the workpile
111  // iterator as the HC implementation caches them.
112  }
113  // In this implementation we expect the caller to have
114  // allocated memory for us since we can return at most one
115  // guid (most likely store using the address of a local)
116  if (NULL_GUID != popped) {
117  *count = 1;
118  *edts = popped;
119  } else {
120  *count = 0;
121  }
122  return 0;
123 }

125 static u8 hcSchedulerGive (ocrScheduler_t* base, u32 count, ocrGuid_t* edts, struct _ocrPolicyCtx_t *context ) {
126  // Source must be a worker guid
127  u64 workerId = context->sourceId;
128  ocrWorkpile_t * wp_to_push = pushMappingOneToOne(base, workerId);
129  u32 i = 0;
130  for ( ; i < count; ++i ) {

132  }

136 static void destructSchedulerHc(ocrScheduler_t * scheduler) {
137  // Free the workpile steal iterator cache
138  ocrSchedulerHc_t * derived = (ocrSchedulerHc_t *) scheduler;
139  u64 workpileCount = scheduler->workpileCount;
140  ocrWorkpileIterator_t ** stealIterators = derived->stealIterators;
141  // Workpiles are deallocated by the policy domain
142  u64 i = 0;
143  while(i < workpileCount) {
144  workpileIteratorDestruct(stealIterators[i]);
145  i++;
146  }
147  free(stealIterators);
148  // free self (workpiles are not allocated by the scheduler)
149  free(scheduler);
150 }
