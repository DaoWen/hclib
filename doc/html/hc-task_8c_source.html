129 static void newTaskHcInternalCommon (ocrPolicyDomain_t * pd, ocrTaskHc_t* derived,
130  ocrTaskTemplate_t * taskTemplate, u32 paramc,
131  u64* paramv, u32 depc, ocrGuid_t outputEvent) {
132  if (depc == 0) {
133  derived->signalers = END_OF_LIST;
134  } else {
135  // Since we know how many dependences we have, preallocate signalers
136  derived->signalers = checkedMalloc(derived->signalers, sizeof(regNode_t)*depc);
137  }
138  derived->waiters = END_OF_LIST;
139  // Initialize base
140  ocrTask_t* base = (ocrTask_t*) derived;
141  base->guid = UNINITIALIZED_GUID;
142  guidify(pd, (u64)base, &(base->guid), OCR_GUID_EDT);
143  base->templateGuid = taskTemplate->guid;
144  base->paramc = paramc;
145  if(paramc) {
146  base->paramv = checkedMalloc(base->paramv, sizeof(u64)*base->paramc);
147  memcpy(base->paramv, paramv, sizeof(u64)*base->paramc);
148  } else {
149  base->paramv = NULL;
150  }
151  base->outputEvent = outputEvent;
152  base->depc = depc;
153  base->addedDepCounter = pd->getAtomic64(pd, NULL /*Context*/);
154  // Initialize ELS
155  int i = 0;
156  while (i < ELS_SIZE) {
157  base->els[i++] = NULL_GUID;
158  }

161 static ocrTaskHc_t* newTaskHcInternal (ocrTaskFactory_t* factory, ocrPolicyDomain_t * pd,
162  ocrTaskTemplate_t * taskTemplate, u32 paramc,
163  u64* paramv, u32 depc, u16 properties,
164  ocrGuid_t affinity, ocrGuid_t outputEvent) {
165  ocrTaskHc_t* newEdt = (ocrTaskHc_t*)checkedMalloc(newEdt, sizeof(ocrTaskHc_t));
166  newTaskHcInternalCommon(pd, newEdt, taskTemplate, paramc, paramv, depc, outputEvent);
167  ocrTask_t * newEdtBase = (ocrTask_t *) newEdt;
168  // If we are creating a finish-edt
169  if (hasProperty(properties, EDT_PROP_FINISH)) {
170  ocrPolicyCtx_t *context = getCurrentWorkerContext();
171 
172  ocrGuid_t latchGuid;
173  pd->createEvent(pd, &latchGuid, OCR_EVENT_FINISH_LATCH_T,
174  false, context);
175  ocrEvent_t *latch = NULL;
176  deguidify(pd, latchGuid, (u64*)&latch, NULL);
177  ocrEventHcFinishLatch_t * hcLatch = (ocrEventHcFinishLatch_t *) latch;
178  // Set the owner of the latch
179  hcLatch->ownerGuid = newEdtBase->guid;
180  ocrEvent_t * parentLatch = getFinishLatch(getCurrentTask());
181  if (parentLatch != NULL) {
182  DPRINTF(DEBUG_LVL_INFO, "Checkin 0x%lx on parent flatch 0x%lx\n", newEdtBase->guid, parentLatch->guid);
183  // Check in current finish latch
184  finishLatchCheckin(parentLatch);
185  // Link the new latch to the parent's decrement slot
186  // When this finish scope is done, the parent scope is signaled.
187  // DESIGN can modify the finish latch to have a standard list of regnode and just addDependence
188  hcLatch->parentLatchWaiter.guid = parentLatch->guid;
189  hcLatch->parentLatchWaiter.slot = OCR_EVENT_LATCH_DECR_SLOT;
190  } else {
191  hcLatch->parentLatchWaiter.guid = NULL_GUID;
192  hcLatch->parentLatchWaiter.slot = -1;
193  }
194  DPRINTF(DEBUG_LVL_INFO, "Checkin 0x%lx on self flatch 0x%lx\n", newEdtBase->guid, latch->guid);
195  // Check in the new finish scope
196  finishLatchCheckin(latch);
197  // Set edt's ELS to the new latch
198  setFinishLatch(newEdtBase, latch->guid);
199  // If there's an output event for this finish edt, add a dependence
200  // from the finish latch to it.
201  // DESIGN can modify flatch to have a standard list of regnode and just addDependence in a if !NULL_GUID
202  hcLatch->outputEventWaiter.guid = outputEvent;
203  hcLatch->outputEventWaiter.slot = 0;
204  } else {
205  // If the currently executing edt is in a finish scope,
206  // but is not a finish-edt itself, just register to the scope
207  ocrEvent_t * curLatch = getFinishLatch(getCurrentTask());
208  if (curLatch != NULL) {
209  DPRINTF(DEBUG_LVL_INFO, "Checkin 0x%lx on current flatch 0x%lx\n", newEdtBase->guid, curLatch->guid);
210  // Check in current finish latch
211  finishLatchCheckin(curLatch);
212  // Transmit finish-latch to newly created edt
213  setFinishLatch(newEdtBase, curLatch->guid);
214  }
215  }
216  return newEdt;
217 }

235 // Signals an edt one of its dependence slot is satisfied
236 static void taskSignaled(ocrTask_t * base, ocrGuid_t data, u32 slot) {
237  // An EDT has a list of signalers, but only register
238  // incrementally as signals arrive.
239  // Assumption: signal frontier is initialized at slot zero
240  // Whenever we receive a signal, it can only be from the
241  // current signal frontier, since it is the only signaler
242  // the edt is registered with at that time.
243  ocrTaskHc_t * self = (ocrTaskHc_t *) base;
244  ocrGuid_t signalerGuid = self->signalers[slot].guid;
245 
246  if (isEventGuidOfKind(signalerGuid, OCR_EVENT_ONCE_T)) {
247  ocrEventHcOnce_t * onceEvent = NULL;
248  deguidify(getCurrentPD(), signalerGuid, (u64*)&onceEvent, NULL);
249  DPRINTF(DEBUG_LVL_INFO, "Decrement ONCE event reference %lx \n", signalerGuid);
250  onceEvent->nbEdtRegistered->fctPtrs->xadd(onceEvent->nbEdtRegistered, -1);
251  if(onceEvent->nbEdtRegistered->fctPtrs->val(onceEvent->nbEdtRegistered) == 0) {
252  // deallocate once event
253  ocrEvent_t * base = (ocrEvent_t *) onceEvent;
254  base->fctPtrs->destruct(base);
255  }
256  }
257  // Replace the signaler's guid by the data guid, this to avoid
258  // further references to the event's guid, which is good in general
259  // and crucial for once-event since they are being destroyed on satisfy.
260  self->signalers[slot].guid = data;
261  if (slot == (base->depc-1)) {
262  // All dependencies have been satisfied, schedule the edt
263  taskSchedule(base->guid);
264  } else {
265  // else register the edt on the next event to wait on
266  slot++;
267  //TODO, could we directly pass the node to avoid a new alloc ?
268  registerWaiter((self->signalers[slot]).guid, base->guid, slot);
269  }
270 }

299 static inline void taskSchedule( ocrGuid_t taskGuid ) {
300  DPRINTF(DEBUG_LVL_INFO, "Schedule 0x%lx\n", taskGuid);
301  // Setting up the context
302  ocrPolicyCtx_t * orgCtx = getCurrentWorkerContext();
303  // Current worker schedulers to current policy domain
304  // TODO this can be cached somewhere if target/dest are static
305  ocrPolicyCtx_t * ctx = orgCtx->clone(orgCtx);
306  ctx->destPD = orgCtx->sourcePD;
307  ctx->destObj = NULL_GUID;
308  ctx->type = PD_MSG_EDT_READY;
309  // give the edt to the policy domain
310  orgCtx->PD->giveEdt(orgCtx->PD, 1, &taskGuid, ctx);
311  ctx->destruct(ctx);
312 }

319 static void tryScheduleTask( ocrTask_t* base ) {
320  //DESIGN This is very much specific to the way we've chosen
321  // to handle event-to-task dependence, which needs some
322  // kind of bootstrapping. This could be done when the last
323  // dependence slot is registered with a signaler, but then
324  // the semantic of the schedule function is weaker.
325  ocrTaskHc_t* self = (ocrTaskHc_t*)base;
326  if (base->depc != 0) {
327  // Register the task on the first dependence. If all
328  // dependences are here, the task is scheduled by
329  // the dependence management code.
330  registerWaiter(self->signalers[0].guid, base->guid, 0);
331  } else {
332  // If there's no dependence, the task must be scheduled now.
333  taskSchedule(base->guid);
334  }
335 }

337 static void taskExecute ( ocrTask_t* base ) {
338  DPRINTF(DEBUG_LVL_INFO, "Execute 0x%lx\n", base->guid);
339  ocrTaskHc_t* derived = (ocrTaskHc_t*)base;
340  // In this implementation each time a signaler has been satisfied, its guid
341  // has been replaced by the db guid it has been satisfied with.
342  u32 paramc = base->paramc;
343  u64 * paramv = base->paramv;
344  u64 depc = base->depc;
345 
346  ocrEdtDep_t * depv = NULL;
347  // If any dependencies, acquire their data-blocks
348  if (depc != 0) {
349  u64 i = 0;
350  //TODO would be nice to resolve regNode into guids before
351  depv = (ocrEdtDep_t *) checkedMalloc(depv, sizeof(ocrEdtDep_t) * depc);
352  // Double-check we're not rescheduling an already executed edt
353  ASSERT(derived->signalers != END_OF_LIST);
354  while ( i < depc ) {
355  //TODO would be nice to standardize that on satisfy
356  regNode_t * regNode = &(derived->signalers[i]);
357  ocrGuid_t dbGuid = regNode->guid;
358  depv[i].guid = dbGuid;
359  if(dbGuid != NULL_GUID) {
360  ASSERT(isDatablockGuid(dbGuid));
361  ocrDataBlock_t * db = NULL;
362  deguidify(getCurrentPD(), dbGuid, (u64*)&db, NULL);
363  depv[i].ptr = db->fctPtrs->acquire(db, base->guid, true);
364  } else {
365  depv[i].ptr = NULL;
366  }
367  i++;
368  }
369  free(derived->signalers);
370  derived->signalers = END_OF_LIST;
371  }
372 
373  ocrTaskTemplate_t * taskTemplate;
374  deguidify(getCurrentPD(), base->templateGuid, (u64*)&taskTemplate, NULL);
375  //TODO: define when task template is resolved from its guid
376  ocrGuid_t retGuid = taskTemplate->executePtr(paramc, paramv,
377  depc, depv);
378 
379  // edt user code is done, if any deps, release data-blocks
380  if (depc != 0) {
381  u64 i = 0;
382  for(i=0; i<depc; ++i) {
383  if(depv[i].guid != NULL_GUID) {
384  ocrDataBlock_t * db = NULL;
385  deguidify(getCurrentPD(), depv[i].guid, (u64*)&db, NULL);
386  RESULT_ASSERT(db->fctPtrs->release(db, base->guid, true), ==, 0);
387  }
388  }
389  free(depv);
390  }
391  if(paramv)
392  free(paramv); // Free the parameter array (we copied them in)
393  bool satisfyOutputEvent = (base->outputEvent != NULL_GUID);
394  // check out from current finish scope
395  ocrEvent_t * curLatch = getFinishLatch(base);
396  if (curLatch != NULL) {
397  // if own the latch, then set the retGuid
398  if (isFinishLatchOwner(curLatch, base->guid)) {
399  ((ocrEventHcFinishLatch_t *) curLatch)->returnGuid = retGuid;
400  satisfyOutputEvent = false;
401  }
402  // If the edt is the last to checkout from the current finish scope,
403  // the latch event automatically satisfies the parent latch (if any)
404  // and the output event associated with the current finish-edt (if any)
405  DPRINTF(DEBUG_LVL_INFO, "Checkout 0x%lx on flatch 0x%lx\n", base->guid, curLatch->guid);
406  finishLatchCheckout(curLatch);
407  }
408 
409  // !! Warning: curLatch might have been deallocated at that point !!
410 
411  if (satisfyOutputEvent) {
412  ocrEvent_t * outputEvent;
413  deguidify(getCurrentPD(), base->outputEvent, (u64*)&outputEvent, NULL);
414  // We know the output event must be of type single sticky since it is
415  // internally allocated by the runtime
416  ASSERT(isEventSingleGuid(base->outputEvent));
417  outputEvent->fctPtrs->satisfy(outputEvent, retGuid, 0);
418  }
419 }

508 // Registers a 'waiter' that should be signaled by 'self' on a particular slot.
509 // If 'self' has already been satisfied, it signals 'waiter' right away.
510 // Warning: Concurrent with event's put.
511 static void awaitableEventRegisterWaiter(ocrEventHcAwaitable_t * self, ocrGuid_t waiter, int slot) {
512  // Try to insert 'waiter' at the beginning of the list
513  regNode_t * curHead = (regNode_t *) self->waiters;
514  if(curHead != SEALED_LIST) {
515  regNode_t * newHead = checkedMalloc(newHead, sizeof(regNode_t));
516  newHead->guid = waiter;
517  newHead->slot = slot;
518  newHead->next = (regNode_t *) curHead;
519  while(curHead != SEALED_LIST && !__sync_bool_compare_and_swap(&(self->waiters), curHead, newHead)) {
520  curHead = (regNode_t *) self->waiters;
521  newHead->next = (regNode_t *) curHead;
522  }
523  if (curHead != SEALED_LIST) {
524  // Insertion successful, we're done
525  DPRINTF(DEBUG_LVL_INFO, "AddDependence from 0x%lx to 0x%lx slot %d\n", (((ocrEvent_t*)self)->guid), waiter, slot);
526  return;
527  }
528  //else list has been sealed by a concurrent satisfy
529  //need to reclaim non-inserted node
530  free(newHead);
531  }
532  // Either the event was satisfied to begin with
533  // or while we were trying to insert the waiter,
534  // the event has been satisfied.
535  signalWaiter(waiter, self->data, slot);
536 }

549 void registerDependence(ocrGuid_t signalerGuid, ocrGuid_t waiterGuid, int slot) {
550  // Warning: signalerGuid can actually be a NULL_GUID.
551  // This can happen when users declared a certain number of
552  // depc but dynamically decide some are not used.
553 
554  // WAIT MODE: event-to-event registration
555  // Note: do not call 'registerWaiter' here as it triggers event-to-edt
556  // registration, which should only be done on edtSchedule.
557  if (isEventGuid(signalerGuid) && isEventGuid(waiterGuid)) {
558  ocrEventHcAwaitable_t * target;
559  deguidify(getCurrentPD(), signalerGuid, (u64*)&target, NULL);
560  awaitableEventRegisterWaiter(target, waiterGuid, slot);
561  return;
562  }
563  // ONCE event must know who are consuming them so that
564  // they are not deallocated prematurely
565  if (isEventGuidOfKind(signalerGuid, OCR_EVENT_ONCE_T) && isEdtGuid(waiterGuid)) {
566  ocrEvent_t * signalerEvent;
567  deguidify(getCurrentPD(), signalerGuid, (u64*)&signalerEvent, NULL);
568  onceEventRegisterEdtWaiter(signalerEvent, waiterGuid, slot);
569  }
570 
571  // SIGNAL MODE:
572  // - anything-to-edt registration
573  // - db-to-event registration
574  registerSignaler(signalerGuid, waiterGuid, slot);
575 }

577 // Registers a waiter on a signaler
578 void registerWaiter(ocrGuid_t signalerGuid, ocrGuid_t waiterGuid, int slot) {
579  if(NULL_GUID == signalerGuid) {
580  // If the dependence was a NULL_GUID, consider this slot is already satisfied
581  signalWaiter(waiterGuid, NULL_GUID, slot);
582  } else if (isEventGuid(signalerGuid)) {
583  ASSERT(isEdtGuid(waiterGuid) || isEventGuid(waiterGuid));
584  ocrEventHcAwaitable_t * target;
585  deguidify(getCurrentPD(), signalerGuid, (u64*)&target, NULL);
586  awaitableEventRegisterWaiter(target, waiterGuid, slot);
587  } else if(isDatablockGuid(signalerGuid) && isEdtGuid(waiterGuid)) {
588  signalWaiter(waiterGuid, signalerGuid, slot);
589  } else {
590  // Everything else is an error
591  ASSERT("error: Unsupported guid kind in registerWaiter" );
592  }
593 }

595 // register a signaler on a waiter
596 void registerSignaler(ocrGuid_t signalerGuid, ocrGuid_t waiterGuid, int slot) {
597  // anything to edt registration
598  if (isEdtGuid(waiterGuid)) {
599  // edt waiting for a signal from an event or a datablock
600  ASSERT((signalerGuid == NULL_GUID) || isEventGuid(signalerGuid) || isDatablockGuid(signalerGuid));
601  ocrTask_t * target = NULL;
602  deguidify(getCurrentPD(), waiterGuid, (u64*)&target, NULL);
603  edtRegisterSignaler(target, signalerGuid, slot);
604  if ( target->depc == target->addedDepCounter->fctPtrs->xadd(target->addedDepCounter,1) ) {
605  // This function pointer is called once, when all the dependence have been added
606  target->fctPtrs->schedule(target);
607  }
608  return;
609  // datablock to event registration => satisfy on the spot
610  } else if (isDatablockGuid(signalerGuid)) {
611  ASSERT(isEventGuid(waiterGuid));
612  ocrEvent_t * target = NULL;
613  deguidify(getCurrentPD(), waiterGuid, (u64*)&target, NULL);
614  // This looks a duplicate of signalWaiter, however there hasn't
615  // been any signal strictly speaking, hence calling satisfy directly
616  ASSERT(isEventSingleGuid(waiterGuid) || isEventLatchGuid(waiterGuid));
617  target->fctPtrs->satisfy(target, signalerGuid, slot);
618  return;
619  }
620  // Remaining legal registrations is event-to-event, but this is
621  // handled in 'registerWaiter'
622  ASSERT(isEventGuid(waiterGuid) && isEventGuid(signalerGuid) && "error: Unsupported guid kind in registerDependence");
623 }

625 // signal a 'waiter' data has arrived on a particular slot
626 void signalWaiter(ocrGuid_t waiterGuid, ocrGuid_t data, int slot) {
627  // TODO do we need to know who's signaling ?
628  if (isEventSingleGuid(waiterGuid)) {
629  ocrEvent_t * target = NULL;
630  deguidify(getCurrentPD(), waiterGuid, (u64*)&target, NULL);
631  singleEventSignaled(target, data, slot);
632  } else if (isEventLatchGuid(waiterGuid)) {
633  ocrEvent_t * target = NULL;
634  deguidify(getCurrentPD(), waiterGuid, (u64*)&target, NULL);
635  latchEventSignaled(target, data, slot);
636  } else if(isEdtGuid(waiterGuid)) {
637  ocrTask_t * target = NULL;
638  deguidify(getCurrentPD(), waiterGuid, (u64*)&target, NULL);
639  taskSignaled(target, data, slot);
640  } else {
641  // ERROR
642  ASSERT(0 && "error: Unsupported guid kind in signal");
643  }
