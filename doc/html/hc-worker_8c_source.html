49 void hcStartWorker(ocrWorker_t * base, ocrPolicyDomain_t * policy) {
50  ocrWorkerHc_t * hcWorker = (ocrWorkerHc_t *) base;
51  hcWorker->run = true;
52 
53  // Starts everybody, the first comp-platform has specific
54  // code to represent the master thread.
55  u64 computeCount = base->computeCount;
56  // What the compute target will execute
57  launchArg_t * launchArg = malloc(sizeof(launchArg_t));
58  launchArg->routine = (hcWorker->id == 0) ? master_worker_computation_routine : worker_computation_routine;
59  launchArg->arg = base;
60  launchArg->PD = policy;
61  u64 i = 0;
62  for(i = 0; i < computeCount; i++) {
63  base->computes[i]->fctPtrs->start(base->computes[i], policy, launchArg);
64  }
65 
66  if (hcWorker->id == 0) {
67  // Worker zero doesn't start the underlying thread since it is
68  // falling through after that start. However, it stills need
69  // to set its local storage data.
70  associate_comp_platform_and_worker(policy, base);
71  }
72 }

137 void worker_loop(ocrPolicyDomain_t * pd, ocrWorker_t * worker) {
138  // Build and cache a take context
139  ocrPolicyCtx_t * orgCtx = getCurrentWorkerContext();
140  ocrPolicyCtx_t * ctx = orgCtx->clone(orgCtx);
141  ctx->type = PD_MSG_EDT_TAKE;
142  // Entering the worker loop
143  while(worker->fctPtrs->isRunning(worker)) {
144  ocrGuid_t taskGuid;
145  u32 count;
146  pd->takeEdt(pd, NULL, &count, &taskGuid, ctx);
147  // remove this when we can take a bunch and make sure there's
148  // an agreement whether it's the callee or the caller that
149  // allocates the taskGuid array
150  ASSERT(count <= 1);
151  if (count != 0) {
152  ocrTask_t* task = NULL;
153  deguidify(pd, taskGuid, (u64*)&(task), NULL);
154  worker->fctPtrs->execute(worker, task, taskGuid, NULL_GUID);
155  task->fctPtrs->destruct(task);
156  }
157  }
158  ctx->destruct(ctx);
159 }

161 void * worker_computation_routine(void * arg) {
162  // Need to pass down a data-structure
163  launchArg_t * launchArg = (launchArg_t *) arg;
164  ocrPolicyDomain_t * pd = launchArg->PD;
165  ocrWorker_t * worker = (ocrWorker_t *) launchArg->arg;
166  // associate current thread with the worker
167  associate_comp_platform_and_worker(pd, worker);
168  // Setting up this worker context to takeEdts
169  // This assumes workers are not relocatable
170  DPRINTF(DEBUG_LVL_INFO, "Starting scheduler routine of worker %d\n", get_worker_id(worker));
171  worker_loop(pd, worker);
172  return NULL;
173 }
