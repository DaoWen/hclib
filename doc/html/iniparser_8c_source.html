577 static line_status iniparser_line(
578  const char * input_line,
579  char * section,
580  char * key,
581  char * value)
582 {
583  line_status sta ;
584  char line[ASCIILINESZ+1];
585  int len ;
586 
587  strcpy(line, strstrip(input_line));
588  len = (int)strlen(line);
589 
590  sta = LINE_UNPROCESSED ;
591  if (len<1) {
592  /* Empty line */
593  sta = LINE_EMPTY ;
594  } else if (line[0]=='#' || line[0]==';') {
595  /* Comment line */
596  sta = LINE_COMMENT ;
597  } else if (line[0]=='[' && line[len-1]==']') {
598  /* Section name */
599  sscanf(line, "[%[^]]", section);
600  strcpy(section, strstrip(section));
601  strcpy(section, strlwc(section));
602  sta = LINE_SECTION ;
603  } else if (sscanf (line, "%[^=] = \"%[^\"]\"", key, value) == 2
604  || sscanf (line, "%[^=] = '%[^\']'", key, value) == 2
605  || sscanf (line, "%[^=] = %[^;#]", key, value) == 2) {
606  /* Usual key=value, with or without comments */
607  strcpy(key, strstrip(key));
608  strcpy(key, strlwc(key));
609  strcpy(value, strstrip(value));
610  /*
611  * sscanf cannot handle '' or "" as empty values
612  * this is done here
613  */
614  if (!strcmp(value, "\"\"") || (!strcmp(value, "''"))) {
615  value[0]=0 ;
616  }
617  sta = LINE_VALUE ;
618  } else if (sscanf(line, "%[^=] = %[;#]", key, value)==2
619  || sscanf(line, "%[^=] %[=]", key, value) == 2) {
620  /*
621  * Special cases:
622  * key=
623  * key=;
624  * key=#
625  */
626  strcpy(key, strstrip(key));
627  strcpy(key, strlwc(key));
628  value[0]=0 ;
629  sta = LINE_VALUE ;
630  } else {
631  /* Generate syntax error */
632  sta = LINE_ERROR ;
633  }
634  return sta ;
635 }

651 dictionary * iniparser_load(const char * ininame)
652 {
653  FILE * in ;
654 
655  char line [ASCIILINESZ+1] ;
656  char section [ASCIILINESZ+1] ;
657  char key [ASCIILINESZ+1] ;
658  char tmp [ASCIILINESZ+1] ;
659  char val [ASCIILINESZ+1] ;
660 
661  int last=0 ;
662  int len ;
663  int lineno=0 ;
664  int errs=0;
665 
666  dictionary * dict ;
667 
668  if ((in=fopen(ininame, "r"))==NULL) {
669  fprintf(stderr, "iniparser: cannot open %s\n", ininame);
670  return NULL ;
671  }
672 
673  dict = dictionary_new(0) ;
674  if (!dict) {
675  fclose(in);
676  return NULL ;
677  }
678 
679  memset(line, 0, ASCIILINESZ);
680  memset(section, 0, ASCIILINESZ);
681  memset(key, 0, ASCIILINESZ);
682  memset(val, 0, ASCIILINESZ);
683  last=0 ;
684 
685  while (fgets(line+last, ASCIILINESZ-last, in)!=NULL) {
686  lineno++ ;
687  len = (int)strlen(line)-1;
688  if (len==0)
689  continue;
690  /* Safety check against buffer overflows */
691  if (line[len]!='\n') {
692  fprintf(stderr,
693  "iniparser: input line too long in %s (%d)\n",
694  ininame,
695  lineno);
696  dictionary_del(dict);
697  fclose(in);
698  return NULL ;
699  }
700  /* Get rid of \n and spaces at end of line */
701  while ((len>=0) &&
702  ((line[len]=='\n') || (isspace(line[len])))) {
703  line[len]=0 ;
704  len-- ;
705  }
706  /* Detect multi-line */
707  if (len>=0 && line[len]=='\\') {
708  /* Multi-line value */
709  last=len ;
710  continue ;
711  } else {
712  last=0 ;
713  }
714  switch (iniparser_line(line, section, key, val)) {
715  case LINE_EMPTY:
716  case LINE_COMMENT:
717  break ;
718 
719  case LINE_SECTION:
720  errs = dictionary_set(dict, section, NULL);
721  break ;
722 
723  case LINE_VALUE:
724  sprintf(tmp, "%s:%s", section, key);
725  errs = dictionary_set(dict, tmp, val) ;
726  break ;
727 
728  case LINE_ERROR:
729  fprintf(stderr, "iniparser: syntax error in %s (%d):\n",
730  ininame,
731  lineno);
732  fprintf(stderr, "-> %s\n", line);
733  errs++ ;
734  break;
735 
736  default:
737  break ;
738  }
739  memset(line, 0, ASCIILINESZ);
740  last=0;
741  if (errs<0) {
742  fprintf(stderr, "iniparser: memory allocation failure\n");
743  break ;
744  }
745  }
746  if (errs) {
747  dictionary_del(dict);
748  dict = NULL ;
749  }
750  fclose(in);
751  return dict ;
752 }
