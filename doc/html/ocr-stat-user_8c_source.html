68  FILTER_TYPE(simple)* rself = (FILTER_TYPE(simple)*)self;

77 static void simpleFilterDestruct(ocrStatsFilter_t *self) {
78  FILTER_TYPE(simple)* rself = (FILTER_TYPE(simple)*)self;
79 
80  DPRINTF(DEBUG_LVL_VERB, "Destroying simple filter @ 0x%lx\n", (u64)self);

92 static u64 simpleFilterDump(ocrStatsFilter_t *self, char** out, u64 chunk, void* configuration) {
93  FILTER_TYPE(simple)* rself = (FILTER_TYPE(simple)*)self;
94 
95  if(rself->count == 0) return 0;
96 
97  ASSERT(chunk < rself->count);
98 
99  ocrGuidKind srcK, destK;
100 
101  *out = (char*)malloc(sizeof(char)*82); // The output message should always fit in 80 chars given the format
102 
103  intSimpleMessageNode_t *tmess = &(rself->messages[chunk]);
104 
105  ocrStat
106  guidKind(getCurrentPD(),tmess->src, &srcK);
107  guidKind(getCurrentPD(),tmess->dest, &destK);
108 
109  snprintf(*out, sizeof(char)*82, "%ld : T %d 0x%lx(%d) -> 0x%lx(%d) ", tmess->tick,
110  tmess->type, tmess->src, srcK, tmess->dest, destK);
111 
112  if(chunk < rself->count - 1)
113  return chunk+1;
114  return 0;
115 }

117 static void simpleFilterNotify(ocrStatsFilter_t *self, const ocrStatsMessage_t *mess) {
118  FILTER_TYPE(simple)* rself = (FILTER_TYPE(simple)*)self;
119  if(rself->count + 1 == rself->maxCount) {
120  // Allocate more space
121  rself->maxCount *= 2;
122  intSimpleMessageNode_t *t = (intSimpleMessageNode_t*)malloc(sizeof(intSimpleMessageNode_t)*rself->maxCount);
123  memcpy(t, rself->messages, (rself->count - 1)*sizeof(intSimpleMessageNode_t));
124  rself->messages = t;
125  }
126  DPRINTF(DEBUG_LVL_VVERB, "Filter @ 0x%lx received message (%ld): %ld -> %ld of type %d\n",
127  (u64)self, rself->count, mess->src, mess->dest, (u32)mess->type);
128  intSimpleMessageNode_t* tmess = &(rself->messages[rself->count++]);
129  tmess->tick = mess->tick;
130  tmess->src = mess->src;
131  tmess->dest = mess->dest;
132  tmess->type = mess->type;
133 }

184 static void fileDumpFilterMerge(ocrStatsFilter_t *self, ocrStatsFilter_t *other, u8 toKill) {
185  // This is when we need to basically dump all information from this filter in the file
186  FILTER_TYPE(filedump)* rself = (FILTER_TYPE(filedump)*)self;
187 
188  DPRINTF(DEBUG_LVL_VERB, "Filter @ 0x%lx received merging from filter 0x%lx\n", (u64)self, (u64)other);
189 
190  if(rself->outFile) {
191  char* outString;
192  u64 nextChunk = 0;
193  rself->lock->lock(rself->lock);
194  do {
195  nextChunk = other->dump(other, &outString, nextChunk, NULL);
196  if(outString) {
197  fprintf(rself->outFile, "%s\n", outString);
198  free(outString);
199  }
200  } while(nextChunk);
201  rself->lock->unlock(rself->lock);
202  }
203 
204  if(toKill) {
205  other->parent = NULL;
206  other->destruct(other);
207  }
208 }
