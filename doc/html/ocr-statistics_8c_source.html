56 void ocrStatsProcessDestruct(ocrStatsProcess_t *self) {
57  u64 i;
58  u32 j;
59 
60  DPRINTF(DEBUG_LVL_INFO, "Destroying StatsProcess 0x%lx (GUID: 0x%lx)\n", (u64)self, self->me);
61 
62  // Make sure we process all remaining messages
63  while(!self->processing->trylock(self->processing)) ;
64 
65  while(intProcessMessage(self)) ;
66 
67  // Free all the filters associated with this process
68  // Assumes that the filter is only associated with ONE
69  // process
70  u32 maxJ = (u32)(self->filterCounts[0] & 0xFFFFFFFF);
71  for(j = 0; j < maxJ; ++j) {
72  self->filters[0][j]->destruct(self->filters[0][j]);
73  }
74  if(maxJ)
75  free(self->filters[0]);
76  for(i = 1; i < (u64)STATS_EVT_MAX + 1; ++i) {
77  if(self->filterCounts[i])
78  free(self->filters[i]);
79  }
80 
81  free(self->filters);
82  free(self->filterCounts);
83 
84  self->processing->unlock(self->processing);
85  self->processing->destruct(self->processing);
86 
87  self->messages->destruct(self->messages);
88 }

90 void ocrStatsProcessRegisterFilter(ocrStatsProcess_t *self, u64 bitMask,
91  ocrStatsFilter_t *filter) {
92 
93  DPRINTF(DEBUG_LVL_VERB, "Registering filter 0x%lx with process 0x%lx for mask 0x%lx\n",
94  (u64)filter, (u64)self, bitMask);
95  u32 countAlloc, countPresent;
96 
97  while(bitMask) {
98  // Setting some filter up
99  u64 bitSet = fls64(bitMask);
100  bitMask &= ~(1ULL << bitSet);
101 
102  ASSERT(bitSet < STATS_EVT_MAX);
103  ++bitSet; // 0 is all filters uniquely
104 
105  // Check to make sure we have enough room to add this filter
106  countPresent = (u32)(self->filterCounts[bitSet] & 0xFFFFFFFF);
107  countAlloc = (u32)(self->filterCounts[bitSet] >> 32);
108  // DPRINTF(DEBUG_LVL_VVERB, "For type %ld, have %d present and %d allocated (from 0x%lx)\n",
109  // bitSet, countPresent, countAlloc, self->filterCounts[bitSet]);
110  if(countAlloc <= countPresent) {
111  // Allocate using an exponential model
112  if(countAlloc)
113  countAlloc *= 2;
114  else
115  countAlloc = 1;
116  ocrStatsFilter_t **newAlloc = (ocrStatsFilter_t**)malloc(sizeof(ocrStatsFilter_t*)*countAlloc);
117  // TODO: This memcpy needs to be replaced. The malloc above as well...
118  memcpy(newAlloc, self->filters[bitSet], countPresent*sizeof(ocrStatsFilter_t*));
119  if(countAlloc != 1)
120  free(self->filters[bitSet]);
121  self->filters[bitSet] = newAlloc;
122  }
123  self->filters[bitSet][countPresent++] = filter;
124 
125  // Set the counter properly again
126  self->filterCounts[bitSet] = ((u64)countAlloc << 32) | (countPresent);
127  // DPRINTF(DEBUG_LVL_VVERB, "Setting final counter for %ld to 0x%lx\n",
128  // bitSet, self->filterCounts[bitSet]);
129  }
130  // Do the same thing for bit 0. Only do it once so we only free things once
131  countPresent = (u32)(self->filterCounts[0] & 0xFFFFFFFF);
132  countAlloc = (u32)(self->filterCounts[0] >> 32);
133  if(countAlloc <= countPresent) {
134  // Allocate using an exponential model
135  if(countAlloc)
136  countAlloc *= 2;
137  else
138  countAlloc = 1;
139  ocrStatsFilter_t **newAlloc = (ocrStatsFilter_t**)malloc(sizeof(ocrStatsFilter_t*)*countAlloc);
140  // TODO: This memcpy needs to be replaced. The malloc above as well...
141  memcpy(newAlloc, self->filters[0], countPresent*sizeof(ocrStatsFilter_t*));
142  if(countAlloc != 1)
143  free(self->filters[0]);
144  self->filters[0] = newAlloc;
145  }
146  self->filters[0][countPresent++] = filter;
147 
148  // Set the counter properly again
149  self->filterCounts[0] = ((u64)countAlloc << 32) | (countPresent);
150 
151 }

153 void ocrStatsAsyncMessage(ocrStatsProcess_t *src, ocrStatsProcess_t *dst,
154  ocrStatsMessage_t *msg) {
155 
156 
157  u64 tickVal = (src->tick += 1);
158  DPRINTF(DEBUG_LVL_VVERB, "Message from 0x%lx with timestamp %ld\n", src->me, tickVal);
159  msg->tick = tickVal;
160  msg->state = 0;
161  ASSERT(msg->src == src->me);
162  ASSERT(msg->dest == dst->me);
163 
164  DPRINTF(DEBUG_LVL_VERB, "Message 0x%lx -> 0x%lx of type %d (0x%lx)\n",
165  src->me, dst->me, (int)msg->type, (u64)msg);
166  // Push the message into the messages queue
167  dst->messages->pushTail(dst->messages, (u64)msg);
168 
169  // Now try to get the lock on processing
170  if(dst->processing->trylock(dst->processing)) {
171  // We grabbed the lock
172  DPRINTF(DEBUG_LVL_VERB, "Message 0x%lx -> 0x%lx: grabbing processing lock\n", src->me, dst->me);
173  u32 count = 5; // TODO: Make configurable
174  while(count-- > 0) {
175  if(!intProcessMessage(dst))
176  break;
177  }
178  DPRINTF(DEBUG_LVL_VERB, "Finished processing messages for 0x%lx\n", dst->me);
179  // Unlock
180  dst->processing->unlock(dst->processing);
181  }
182 }

184 void ocrStatsSyncMessage(ocrStatsProcess_t *src, ocrStatsProcess_t *dst,
185  ocrStatsMessage_t *msg) {
186 
187  u64 tickVal = (src->tick += 1);
188  DPRINTF(DEBUG_LVL_VVERB, "SYNC Message from 0x%lx with timestamp %ld\n", src->me, tickVal);
189  msg->tick = tickVal;
190  msg->state = 1;
191  ASSERT(msg->src == src->me);
192  ASSERT(msg->dest == dst->me);
193 
194  DPRINTF(DEBUG_LVL_VERB, "SYNC Message 0x%lx -> 0x%lx of type %d (0x%lx)\n",
195  src->me, dst->me, (int)msg->type, (u64)msg);
196  // Push the message into the messages queue
197  dst->messages->pushTail(dst->messages, (u64)msg);
198 
199  // Now try to get the lock on processing
200  // Since this is a sync message, we need to make sure
201  // we don't come out of this without having had our message
202  // processed. We are going to repeatedly try to grab the lock
203  // with a backoff and sit around until we either process our
204  // message or someone else does it
205  while(1) {
206  if(dst->processing->trylock(dst->processing)) {
207  // We grabbed the lock
208  DPRINTF(DEBUG_LVL_VERB, "Message 0x%lx -> 0x%lx: grabbing processing lock\n", src->me, dst->me);
209  s32 count = 5; // TODO: Make configurable
210  // Process at least count and at least until we get to our message
211  // EXTREMELY RARE PROBLEM of count running over for REALLY deep queues
212  while(count-- > 0 || msg->state != 2) {
213  if(!intProcessMessage(dst) && (msg->state == 2))
214  break;
215  }
216  DPRINTF(DEBUG_LVL_VERB, "Finished processing messages for 0x%lx\n", dst->me);
217  // Unlock
218  dst->processing->unlock(dst->processing);
219 
220  break; // Break out of while(1)
221  } else {
222  // Backoff (TODO, need HW support (abstraction though))
223  }
224  }
225 
226  // At this point, we can sync our own tick
227  ASSERT(msg->state == 2);
228  ASSERT(msg->tick >= src->tick);
229  src->tick = msg->tick;
230  DPRINTF(DEBUG_LVL_VVERB, "Sync tick out: %ld\n", src->tick);
231  msg->destruct(msg);
232 }

248 static u8 intProcessMessage(ocrStatsProcess_t *dst) {
249  ocrStatsMessage_t* msg = (ocrStatsMessage_t*)dst->messages->popHead(dst->messages);
250 
251  if(msg) {
252  DPRINTF(DEBUG_LVL_VERB, "Processing message 0x%lx for 0x%lx\n",
253  (u64)msg, dst->me);
254  u64 newTick = msg->tick > (dst->tick + 1)?msg->tick:(dst->tick + 1);
255  msg->tick = dst->tick = newTick;
256  DPRINTF(DEBUG_LVL_VVERB, "Message tick is %ld\n", msg->tick);
257 
258  u32 type = (u32)msg->type;
259  u32 countFilter = dst->filterCounts[type] & 0xFFFFFFFF;
260  if(countFilter) {
261  // We have at least one filter that is registered to
262  // this message type
263  ocrStatsFilter_t **myFilters = dst->filters[type];
264  while(countFilter-- > 0) {
265  DPRINTF(DEBUG_LVL_VVERB, "Sending message 0x%lx to filter 0x%lx\n",
266  (u64)msg, (u64)myFilters[countFilter]);
267  myFilters[countFilter]->notify(myFilters[countFilter], msg);
268  }
269  }
270  if(msg->state == 1) {
271  msg->state = 2;
272  } else {
273  ASSERT(msg->state == 0);
274  msg->destruct(msg);
275  }
276  return 1;
277  } else {
278  return 0;
279  }
