44 static void * pthreadRoutineWrapper(void * arg) {
45  ocrCompPlatformPthread_t * pthreadCompPlatform = (ocrCompPlatformPthread_t *) arg;
46  launchArg_t * launchArg = (launchArg_t *) pthreadCompPlatform->launchArg;
47  // Wrapper routine to allow initialization of local storage
48  // before entering the worker routine.
49  perThreadStorage_t *data = (perThreadStorage_t*)checkedMalloc(data, sizeof(perThreadStorage_t));
50  RESULT_ASSERT(pthread_setspecific(selfKey, data), ==, 0);
51  if (launchArg != NULL) {
52  return pthreadRoutineExecute(launchArg);
53  }
54  return NULL;
55 }

57 static void pthreadStart(ocrCompPlatform_t * compPlatform, ocrPolicyDomain_t * PD, launchArg_t * launchArg) {
58  ocrCompPlatformPthread_t * pthreadCompPlatform = (ocrCompPlatformPthread_t *) compPlatform;
59  pthreadCompPlatform->launchArg = launchArg;
60  pthread_attr_t attr;
61  RESULT_ASSERT(pthread_attr_init(&attr), ==, 0);
62  //Note this call may fail if the system doesn't like the stack size asked for.
63  RESULT_ASSERT(pthread_attr_setstacksize(&attr, pthreadCompPlatform->stackSize), ==, 0);
64  RESULT_ASSERT(pthread_create(&(pthreadCompPlatform->osThread),
65  &attr, &pthreadRoutineWrapper,
66  pthreadCompPlatform), ==, 0);
67 }

98 static ocrCompPlatform_t* newCompPlatformPthread(ocrCompPlatformFactory_t *factory,
99  ocrParamList_t *perInstance) {
100 
101  pthread_once(&selfKeyInitialized, initializeKey);
102  ocrCompPlatformPthread_t * compPlatformPthread = checkedMalloc(
103  compPlatformPthread, sizeof(ocrCompPlatformPthread_t));
104 
105  paramListCompPlatformPthread_t * params =
106  (paramListCompPlatformPthread_t *) perInstance;
107  if ((params != NULL) && (params->isMasterThread)) {
108  // This particular instance is the master thread
109  ocrCompPlatformFactoryPthread_t * pthreadFactory =
110  (ocrCompPlatformFactoryPthread_t *) factory;
111  compPlatformPthread->base.fctPtrs = &(pthreadFactory->masterPlatformFcts);
112  } else {
113  // This is a regular thread, get regular function pointers
114  compPlatformPthread->base.fctPtrs = &(factory->platformFcts);
115  }
116  compPlatformPthread->stackSize = ((params != NULL) && (params->stackSize > 0)) ? params->stackSize : 8388608;
117  compPlatformPthread->base.module.mapFct = NULL;
118  return (ocrCompPlatform_t*)compPlatformPthread;
119 }
