32 void* regularAcquire(ocrDataBlock_t *self, ocrGuid_t edt, bool isInternal) {
33  ocrDataBlockRegular_t *rself = (ocrDataBlockRegular_t*)self;
34 
35  DPRINTF(DEBUG_LVL_VERB, "Acquiring DB @ 0x%"PRIx64" (GUID: 0x%"PRIdPTR") from EDT 0x%"PRIdPTR" (isInternal %d)\n",
36  (u64)self->ptr, rself->base.guid, edt, (u32)isInternal);
37 
38  // Critical section
39  rself->lock->fctPtrs->lock(rself->lock);
40  if(rself->attributes.freeRequested) {
41  rself->lock->fctPtrs->unlock(rself->lock);
42  return NULL;
43  }
44  u32 idForEdt = ocrGuidTrackerFind(&(rself->usersTracker), edt);
45  if(idForEdt > 63)
46  idForEdt = ocrGuidTrackerTrack(&(rself->usersTracker), edt);
47  else {
48  DPRINTF(DEBUG_LVL_VVERB, "EDT already had acquired DB (pos %d)\n", idForEdt);
49  rself->lock->fctPtrs->unlock(rself->lock);
50  return self->ptr;
51  }
52 
53  if(idForEdt > 63) {
54  rself->lock->fctPtrs->unlock(rself->lock);
55  return NULL;
56  }
57  rself->attributes.numUsers += 1;
58  if(isInternal)
59  rself->attributes.internalUsers += 1;
60 
61  rself->lock->fctPtrs->unlock(rself->lock);
62  // End critical section
63  DPRINTF(DEBUG_LVL_VERB, "Added EDT GUID 0x%"PRIx64" at position %d. Have %d users and %d internal\n",
64  (u64)edt, idForEdt, rself->attributes.numUsers, rself->attributes.internalUsers);
65  return self->ptr;
66 }

68 u8 regularRelease(ocrDataBlock_t *self, ocrGuid_t edt,
69  bool isInternal) {
70 
71  ocrDataBlockRegular_t *rself = (ocrDataBlockRegular_t*)self;
72  u32 edtId = ocrGuidTrackerFind(&(rself->usersTracker), edt);
73  bool isTracked = true;
74 
75  DPRINTF(DEBUG_LVL_VERB, "Releasing DB @ 0x%"PRIx64" (GUID 0x%"PRIdPTR") from EDT 0x%"PRIdPTR" (%d) (internal: %d)\n",
76  (u64)self->ptr, rself->base.guid, edt, edtId, (u32)isInternal);
77  // Start critical section
78  rself->lock->fctPtrs->lock(rself->lock);
79  if(edtId > 63 || rself->usersTracker.slots[edtId] != edt) {
80  // We did not find it. The runtime may be
81  // re-releasing it
82  if(isInternal) {
83  // This is not necessarily an error
84  rself->attributes.internalUsers -= 1;
85  isTracked = false;
86  } else {
87  // Definitely a problem here
88  rself->lock->fctPtrs->unlock(rself->lock);
89  return (u8)EACCES;
90  }
91  }
92 
93  if(isTracked) {
94  ocrGuidTrackerRemove(&(rself->usersTracker), edt, edtId);
95  rself->attributes.numUsers -= 1;
96  if(isInternal) {
97  rself->attributes.internalUsers -= 1;
98  }
99  }
100  DPRINTF(DEBUG_LVL_VVERB, "DB attributes: numUsers %d; internalUsers %d; freeRequested %d\n",
101  rself->attributes.numUsers, rself->attributes.internalUsers, rself->attributes.freeRequested);
102  // Check if we need to free the block
103  if(rself->attributes.numUsers == 0 &&
104  rself->attributes.internalUsers == 0 &&
105  rself->attributes.freeRequested == 1) {
106  // We need to actually free the data-block
107  rself->lock->fctPtrs->unlock(rself->lock);
108  self->fctPtrs->destruct(self);
109  } else {
110  rself->lock->fctPtrs->unlock(rself->lock);
111  }
112  // End critical section
113 
114  return 0;
115 }

117 void regularDestruct(ocrDataBlock_t *self) {
118  // We don't use a lock here. Maybe we should
119  ocrDataBlockRegular_t *rself = (ocrDataBlockRegular_t*)self;
120  ASSERT(rself->attributes.numUsers == 0);
121  ASSERT(rself->attributes.freeRequested == 1);
122  rself->lock->fctPtrs->destruct(rself->lock);
123 
124  ocrPolicyDomain_t *pd = getCurrentPD();
125  ocrPolicyCtx_t *orgCtx = getCurrentWorkerContext();
126  ocrPolicyCtx_t * ctx = orgCtx->clone(orgCtx);
127  ctx->type = PD_MSG_GUID_REL;
128  // Tell the allocator to free the data-block
129  ocrAllocator_t *allocator = NULL;
130  deguidify(getCurrentPD(), rself->base.allocator, (u64*)&allocator, NULL);
131 
132  DPRINTF(DEBUG_LVL_VERB, "Freeing DB @ 0x%"PRIx64" (GUID: 0x%"PRIdPTR")\n", (u64)self->ptr, rself->base.guid);
133  allocator->fctPtrs->free(allocator, self->ptr);
134 
135  // TODO: This is not pretty to be here but I can't put this in the ocrDbFree because
136  // the semantics of ocrDbFree is that it will wait for all acquire/releases to have
137  // completed so I have to release the block when it is really being freed. This current
138  // OCR version does not really implement delayed freeing but may in the future.
139 #ifdef OCR_ENABLE_STATISTICS
140  ocrStatsProcessDestruct(&(rself->base.statProcess));
141 #endif
142 
143  pd->inform(pd, self->guid, ctx);
144  ctx->destruct(ctx);
145  free(rself);
146 }

148 u8 regularFree(ocrDataBlock_t *self, ocrGuid_t edt) {
149  ocrDataBlockRegular_t *rself = (ocrDataBlockRegular_t*)self;
150 
151  u32 id = ocrGuidTrackerFind(&(rself->usersTracker), edt);
152  DPRINTF(DEBUG_LVL_VERB, "Requesting a free for DB @ 0x%"PRIx64" (GUID 0x%"PRIdPTR")\n",
153  (u64)self->ptr, rself->base.guid);
154  // Begin critical section
155  rself->lock->fctPtrs->lock(rself->lock);
156  if(rself->attributes.freeRequested) {
157  rself->lock->fctPtrs->unlock(rself->lock);
158  return EPERM;
159  }
160  rself->attributes.freeRequested = 1;
161  rself->lock->fctPtrs->unlock(rself->lock);
162  // End critical section
163 
164 
165  if(id < 64) {
166  regularRelease(self, edt, false);
167  } else {
168  // We can call free without having acquired the block
169  // Now check if we can actually free the block
170 
171  // Critical section
172  rself->lock->fctPtrs->lock(rself->lock);
173  if(rself->attributes.numUsers == 0 && rself->attributes.internalUsers == 0) {
174  rself->lock->fctPtrs->unlock(rself->lock);
175  regularDestruct(self);
176  } else {
177  rself->lock->fctPtrs->unlock(rself->lock);
178  }
179  // End critical section
180  }
181 
182  return 0;
183 }
