20 static void newTaskFsimInternal (ocrPolicyDomain_t * pd, ocrTaskHc_t* derived,
21  ocrTaskTemplate_t * taskTemplate, u64 * params, void** paramv, ocrGuid_t outputEvent) {
22  u32 nbDeps = taskTemplate->depc;
23  if (nbDeps == 0) {
24  derived->signalers = END_OF_LIST;

30  // Initialize base

34  if (taskTemplate != NULL) {
35  base->templateGuid = taskTemplate->guid;
36  }
37  base->params = params;
38  base->paramv = paramv;
39  base->outputEvent = outputEvent;
40  // Initialize ELS
41  int i = 0;
42  while (i < ELS_SIZE) {

161 void * xe_worker_computation_routine (void * arg) {
162  ocrWorker_t * baseWorker = (ocrWorker_t *) arg;
163 
164  /* associate current thread with the worker */
165  associate_comp_platform_and_worker(baseWorker);
166 
167  ocrGuid_t workerGuid = get_worker_guid(baseWorker);
168  ocrScheduler_t * xeScheduler = get_worker_scheduler(baseWorker);
169 
170  log_worker(INFO, "Starting scheduler routine of worker %d\n", get_worker_id(baseWorker));
171 
172  while ( baseWorker->fctPtrs->isRunning(baseWorker) ) {
173  //
174  // try to extract work from an XE's workpile of assigned-by-CE-tasks
175  // as of now always of size 1
176  ocrGuid_t taskGuid = xeScheduler->take(xeScheduler, workerGuid);
177 
178  if ( NULL_GUID != taskGuid ) {
179  // if managed to find work that was assigned by CE, execute it
180  ocrTask_t* currTask = NULL;
181  deguidify(getCurrentPD(), taskGuid, (u64*)&(currTask), NULL);
182  baseWorker->fctPtrs->setCurrentEDT(baseWorker,taskGuid);
183  currTask->fctPtrs->execute(currTask);
184  baseWorker->fctPtrs->setCurrentEDT(baseWorker, NULL_GUID);
185  } else {
186  // TODO sagnak, this assumes (*A LOT*) the structure below, is this fair?
187  // no assigned work found, now we have to create a 'message task'
188  // by using our policy domain's message task factory
189  ocrPolicyDomainFsim_t* policy_domain = (ocrPolicyDomainFsim_t *) xeScheduler->domain;
190  ocrTaskFactory_t* message_task_factory = policy_domain->taskMessageFactory;
191  // the message to the CE says 'give me work' and notes who is asking for it
192  ocrTaskFsimMessage_t * derived = (ocrTaskFsimMessage_t *)
193  message_task_factory->instantiate(message_task_factory, NULL, NULL, NULL, 0, NULL);
194  derived -> type = GIVE_ME_WORK;
195  derived -> from_worker_guid = workerGuid;
196 
197  // there is no work left and the message has been sent, so turn of the worker
198  // baseWorker->stop(baseWorker);
199  ocrWorkerFsimXE_t* derivedWorker = (ocrWorkerFsimXE_t*)baseWorker;
200  pthread_mutex_lock(&derivedWorker->isRunningMutex);
201  if ( baseWorker->fctPtrs->isRunning(baseWorker) ) {
202  ocrGuid_t messageTaskGuid = derived->fsimBase.base.base.guid;
203  // give the work to the XE scheduler, which in turn should give it to the CE
204  // through policy domain hand out and the scheduler differentiates tasks by type (RTTI) like
205  xeScheduler->give(xeScheduler, workerGuid, messageTaskGuid);
206  pthread_cond_wait(&derivedWorker->isRunningCond, &derivedWorker->isRunningMutex);
207  }
208  pthread_mutex_unlock(&derivedWorker->isRunningMutex);
209  }
210  }
211  return NULL;
212 }

214 void * ce_worker_computation_routine(void * arg) {
215  ocrWorker_t * ceWorker = (ocrWorker_t *) arg;
216  ocrScheduler_t * ceScheduler = get_worker_scheduler(ceWorker);
217 
218  /* associate current thread with the worker */
219  associate_comp_platform_and_worker(ceWorker);
220 
221  ocrGuid_t ceWorkerGuid = get_worker_guid(ceWorker);
222  log_worker(INFO, "Starting scheduler routine of worker %d\n", get_worker_id(ceWorker));
223  while(ceWorker->fctPtrs->isRunning(ceWorker)) {
224 
225  // pop a 'message task' to handle messaging
226  // the 'state' of the scheduler should point to the message workpile
227  ocrGuid_t messageTaskGuid = ceScheduler->take(ceScheduler, ceWorkerGuid);
228 
229  // TODO sagnak, oh the assumptions, clean below
230  if ( NULL_GUID != messageTaskGuid ) {
231 
232  // check if the popped task is of 'message task' type
233  ocrTaskFsimBase_t* currTask = NULL;
234  deguidify(getCurrentPD(), messageTaskGuid, (u64*)&(currTask), NULL);
235 
236  fsim_message_interface_t* taskAsMessage = &(currTask->message_interface);
237  assert(taskAsMessage->is_message(taskAsMessage) && "hoping to pop a message-task; popped non-message task");
238 
239  // TODO sagnak, we seem to 'know' the underlying task type, not sure if this is a good idea
240  ocrTaskFsimMessage_t* currMessageTask = (ocrTaskFsimMessage_t*) currTask;
241 
242  // TODO sagnak, we seem to 'know' the underlying scheduler type, not sure if this is a good idea
243  ocrSchedulerFsimCE_t * derivedCeScheduler = (ocrSchedulerFsimCE_t*) ceScheduler;
244 
245  // find the worker, scheduler the message originated from
246  ocrGuid_t targetWorkerGuid = currMessageTask->from_worker_guid;
247  ocrWorker_t* targetWorker= NULL;
248  deguidify(getCurrentPD(), targetWorkerGuid, (u64*)&(targetWorker), NULL);
249  ocrScheduler_t* targetScheduler = get_worker_scheduler(targetWorker);
250 
251  switch (currMessageTask->type) {
252  case GIVE_ME_WORK:
253  {
254  // if the XE is asking for work
255  // change the scheduler 'state' to 'executable task' popping
256  derivedCeScheduler -> in_message_popping_mode = 0;
257  // pop an 'executable task'
258  ocrGuid_t toXeTaskGuid = ceScheduler->take(ceScheduler, ceWorkerGuid);
259  // change the scheduler 'state' back to 'message task' popping
260  derivedCeScheduler -> in_message_popping_mode = 1;
261 
262  if ( NULL_GUID != toXeTaskGuid ) {
263  targetScheduler->give(targetScheduler, ceWorkerGuid, toXeTaskGuid);
264  // now that the XE has work, it may be restarted
265  // targetWorker->start(targetWorker);
266  //
267  ocrWorkerFsimXE_t* derivedWorker = (ocrWorkerFsimXE_t*)targetWorker;
268  pthread_mutex_lock(&derivedWorker->isRunningMutex);
269  pthread_cond_signal(&derivedWorker->isRunningCond);
270  pthread_mutex_unlock(&derivedWorker->isRunningMutex);
271  } else {
272  // if there was no successful task handing to XE, do not lose the message
273  ceScheduler->give(ceScheduler, ceWorkerGuid, messageTaskGuid);
274  }
275  }
276  break;
277  case PICK_MY_WORK_UP:
278  {
279  ocrGuid_t taskFromXe = targetScheduler->take( targetScheduler, ceWorkerGuid );
280  if ( NULL_GUID != taskFromXe ) {
281  ceScheduler->give(ceScheduler, ceWorkerGuid, taskFromXe);
282  } else {
283  // if there was no successful task picking from XE, do not lose the message
284  ceScheduler->give(ceScheduler, ceWorkerGuid, messageTaskGuid);
285  }
286  }
287  break;
288  };
