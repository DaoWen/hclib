52 /*
53  * VERY IMPORTANT: The maximum size of the memory supported
54  * is *very* dependent on a few parameters:
55  * - tlsfSize_t: this encodes the "addresses" and size of
56  * each block. It must be able to contain the entire range
57  * of addresses
58  * - this is partly compensated by ELEMENT_SIZE_LOG2/BYTES which
59  * contains the number of bytes per "unit" of memory. This
60  * increases granularity to be able to address more with fewer
61  * bits. Therefore, tlsfSize_t needs only to encode
62  * the total size in ELEMENT_SIZE_BYTES unit.
63  * - FL_MAX_LOG2: The total size of memory is (1<<FL_MAX_LOG2) * ELEMENT_SIZE_BYTES
64  */

73  * NOTE: Must be at least 2 bytes

193 /*
194  * Some computed values:
195  * - SL_COUNT: Number of buckets in each SL list
196  * - FL_COUNT_SHIFT: For the first SL_COUNT, we do not need to maintain
197  * first level lists (since they all go in the "0th" one.
198  * - FL_COUNT: Number of buckets in FL (so number of SL bitmaps)

202  SL_COUNT = (1 << SL_COUNT_LOG2),
203  FL_COUNT_SHIFT = (SL_COUNT_LOG2),
204  FL_COUNT = (FL_MAX_LOG2 - FL_COUNT_SHIFT + 1),
205  ZERO_LIST_SIZE = (1 << FL_COUNT_SHIFT),

208 /*
209  * Header of a block (allocated or free) of memory. Note that not all
210  * fields are valid at all times:
211  *
212  * Free block:
213  * tlsfSize_t prevFreeBlock; <-- can NEVER be 0 or 1 (would indicate
214  * an occupied block). This should be fine as that would mean that
215  * the address of the previous block is 0x4 (in absolute) which is
216  * never going to happen...
217  * tlsfSize_t sizeBlock;
218  * tlsfSize_t nextFreeBlock; <-- This field aligned 2^ALIGN_LOG2 bytes
219  * <SPACE>
220  * tlsfSize_t sizeBlock; (used by next occupied block); must be at the very end
221  *
222  * Used block:
223  * tlsfSize_t prevFreeBlock <- not used as that:
224  * - bit 0 indicates whether the previous block is free (1) or not (0)
225  * - the rest MUST be 0 to indicate an occupied block.
226  * tlsfSize_t sizeBlock
227  *
228  * Note that prevFreeBlock and nextFreeBlock are actually pointers
229  * but we only consider the lower tlsfSize_t bits. The rest are zero
230  * since the total size of the memory is small.
231  * */
232 typedef struct {
233  tlsfSize_t prevFreeBlock; // __attribute__ (( aligned (ALIGN_BLOCK_BYTES) ));
234  tlsfSize_t sizeBlock;
235 
236  tlsfSize_t nextFreeBlock; // __attribute__ (( aligned (ALIGN_BYTES) ));
237 } header_t;

253 /* The pool (containing the bitmaps and the lists of
254  * free blocks.
255  */
256 typedef struct {
257  tlsfSize_t flAvailOrNot; // bitmap that indicates the presence
258  // (1) or absence (0) of free blocks
259  // in blocks[i][*]
260  tlsfSize_t slAvailOrNot[FL_COUNT]; // Second level bitmaps
261  tlsfSize_t blocks[FL_COUNT][SL_COUNT]; // Pointers to starts of free-lists
262 
263  // The padding works as follows:
264  // - ((FL_COUNT*(1+SL_COUNT) + 1)*(SIZE_TLSF_SIZE/8)) is the size of what is above in bytes
265  // + Call this quantity x
266  // - we know that x = k*ALIGN_BYTES + r and we want x + p = (k+1)*ALIGN_BYTES
267  // - Therefore p = ALIGN_BYTES - r except when r == 0 in which case p = 0
268  // r is (x & (ALIGN_BYTES - 1)
269 #define _R_SIZE_TEMP (((FL_COUNT*(1+SL_COUNT) + 1)*(SIZE_TLSF_SIZE/8)) & (ALIGN_BYTES - 1))
270  u8 _padding[_R_SIZE_TEMP?(ALIGN_BYTES - _R_SIZE_TEMP):0];
271 #undef _R_SIZE_TEMP
272 
273  /* nullBlock will be used to mark NULL blocks. This is important
274  * since prev/next blocks can never be NULL as that is also used
275  * as the busy flag
276  */
277  header_t nullBlock; // Contains 3*tlsfSize_t
278 #define _R_SIZE_TEMP ((3*(SIZE_TLSF_SIZE/8)) & (ALIGN_BYTES - 1))
279  u8 _padding2[_R_SIZE_TEMP?(ALIGN_BYTES - _R_SIZE_TEMP):0];
280 #undef _R_SIZE_TEMP
281  header_t mainBlock; // Main memory block (this needs to be aligned ALIGN_BYTES)
282 } pool_t;

380 static void linkFreeBlocks(u64 pgStart, headerAddr_t first, headerAddr_t second) {
381 
382  ASSERT(isBlockFree(GET_ADDRESS(first))); /* linkFreeBlocks arg1 not free */
383  ASSERT(isBlockFree(GET_ADDRESS(second))); /* linkFreeBlocks arg2 not free */
384 
385  /* Consec blocks cannot be free */
386  ASSERT(getNextBlock(pgStart, GET_ADDRESS(first)).value != second.value);
387 
388  /* linkFreeBlocks arg1 misaligned */
389  ASSERT((GET_ADDRESS(first) & ((1<<ELEMENT_SIZE_LOG2) - 1)) == 0);
390 
391  /* linkFreeBlocks arg2 misaligned */
392  ASSERT((GET_ADDRESS(second) & ((1<<ELEMENT_SIZE_LOG2) - 1)) == 0);
393 
394  ST_SIZE(ADDR_OF(header_t, GET_ADDRESS(first), nextFreeBlock), second.value);
395  ST_SIZE(ADDR_OF(header_t, GET_ADDRESS(second), prevFreeBlock), first.value);
396  FENCE_STORE;
397 }

423 static void markBlockFree(u64 pgStart, u64 me /* header_t* */) {
424  // To mark a block as free, we need to duplicate it's size
425  // at the end of the block so that the next block can find the beginning
426  // of the free block.
427  tlsfSize_t temp;
428  LD_SIZE(temp, ADDR_OF(header_t, me, sizeBlock));
429  FENCE_LOAD;
430  u64 locationToWrite = me +
431  ((temp + GusedBlockOverhead) << ELEMENT_SIZE_LOG2) - sizeof(tlsfSize_t);
432  ST_SIZE(locationToWrite, temp);
433 
434  ST_SIZE(ADDR_OF(header_t, me, prevFreeBlock), 0xBEEF); // Some value that is not 0 or 1 for now
435  // will be updated when this
436  // free block is actually inserted
437 
438  u64 nextBlockPtr = GET_ADDRESS(getNextBlock(pgStart, me));
439  FENCE_STORE;
440 
441  if(!isBlockFree(nextBlockPtr)) // Usually, no two free blocks follow each other but it may happen temporarily
442  markPrevBlockFree(nextBlockPtr);
443 }

449 static tlsfSize_t getRealSizeOfRequest(u64 size) {
450  if((size >> ELEMENT_SIZE_LOG2) > GmaxBlockRealSize)
451  return 0; // Too big to allocate
452 
453  if(size < ((GminBlockRealSize - GusedBlockOverhead) << ELEMENT_SIZE_LOG2)) {
454  return GminBlockRealSize - GusedBlockOverhead;
455  }
456 
457  // We want this to be aligned so that the total w/ the header is ALIGN_BYTES but
458  // since the header may be smaller than ALIGN_BYTES, we do this weird computation
459  // to take it into account
460  size = ((size + ALIGN_BYTES - GoffsetToStart - 1) & ~(ALIGN_BYTES - 1))
461  + GoffsetToStart;
462 
463  ASSERT(size % ELEMENT_SIZE_BYTES == 0);
464 
465  size >>= ELEMENT_SIZE_LOG2;
466  ASSERT(size >= (GminBlockRealSize - GusedBlockOverhead));
467 
468  return size;
469 }

471 /* Two-level function to determine indices. This is pretty much
472  * taken straight from the specs

474 static void mappingInsert(tlsfSize_t realSize, int* flIndex, int* slIndex) {
475  int tf, ts;
476 
477  if(realSize < ZERO_LIST_SIZE) {
478  tf = 0;
479  ts = realSize;
480  } else {
481  tf = FLS(realSize);
482  ts = (realSize >> (tf - SL_COUNT_LOG2)) - (SL_COUNT);
483  tf -= (FL_COUNT_SHIFT - 1);
484  }
485  *flIndex = tf;
486  *slIndex = ts;
487 }

489 /* For allocations, we want to round-up to the next block size
490  * so that we are sure that any block will work so we can
491  * pick it in constant time.
492  */
493 static void mappingSearch(tlsfSize_t realSize, int* flIndex, int* slIndex) {
494  if(realSize >= ZERO_LIST_SIZE) {
495  // If not, we don't need to do anything since the mappingInsert
496  // will return the correct thing
497  realSize += (1 << (FLS(realSize) - SL_COUNT_LOG2)) - 1;
498  }
499  mappingInsert(realSize, flIndex, slIndex);
500 }

502 /* Search for a suitable free block:
503  * - first search for a block in the fl/sl indicated.
504  * - if not found, search for higher sl (bigger) with same fl
505  * - if not found, search for higher fl and sl in that
506  *
507  * Returns the header of a free block as well as flIndex and slIndex
508  * that block was taken from.
509  */
510 static headerAddr_t findFreeBlockForRealSize(u64 pgStart, tlsfSize_t realSize,
511  int* flIndex, int* slIndex) {
512 
513  int tf, ts;
514  headerAddr_t res = { .address = 0ULL };
515  tlsfSize_t flBitMap, slBitMap;
516 
517  mappingSearch(realSize, flIndex, slIndex);
518  tf = *flIndex;
519  ts = *slIndex;
520 
521  if(tf >= FL_COUNT) {
522  SET_NULL(res); // This means the alloc size is too large
523  return res;
524  }
525 
526  LD_SIZE(slBitMap, ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, tf));
527  FENCE_LOAD;
528  slBitMap &= (~0 << ts); // This takes all SL bins bigger or requal to ts
529  if(slBitMap == 0) {
530  // We don't have any non-zero block here so we look at the flAvailOrNot map
531  LD_SIZE(flBitMap, ADDR_OF(pool_t, pgStart, flAvailOrNot));
532  FENCE_LOAD;
533  flBitMap &= (~0 << (tf + 1));
534  if(flBitMap == 0) {
535  SET_NULL(res);
536  return res;
537  }
538 
539  // Look for the first bit that is a one
540  tf = myffs(flBitMap);
541  ASSERT(tf > *flIndex);
542  *flIndex = tf;
543 
544  // Now we get the slBitMap. There is definitely a 1 in there since there is a 1 in tf
545  LD_SIZE(slBitMap, ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, tf));
546  FENCE_LOAD;
547  }
548 
549  ASSERT(slBitMap != 0);
550 
551  ts = myffs(slBitMap);
552  *slIndex = ts;
553 
554  LD_SIZE(res.value, ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, tf, SL_COUNT, ts));
555  FENCE_LOAD;
556  return res;
557 }

560 /* Helpers to manipulate the free lists */
561 static void removeFreeBlock(u64 pgStart, headerAddr_t freeBlock) {
562  /* header_t* */ u64 freeBlockPtr = GET_ADDRESS(freeBlock);
563  tlsfSize_t tempSz;
564 
565  ASSERT(isBlockFree(freeBlockPtr));
566 
567  int flIndex, slIndex;
568  LD_SIZE(tempSz, ADDR_OF(header_t, freeBlockPtr, sizeBlock));
569  FENCE_LOAD;
570  mappingInsert(tempSz, &flIndex, &slIndex);
571 
572  // First remove it from the chain of free blocks
573  headerAddr_t prevFreeBlock = getPrevFreeBlock(freeBlockPtr);
574  headerAddr_t nextFreeBlock = getNextFreeBlock(freeBlockPtr);
575 
576  ASSERT(prevFreeBlock.value !=0 && isBlockFree(GET_ADDRESS(prevFreeBlock)));
577  ASSERT(nextFreeBlock.value !=0 && isBlockFree(GET_ADDRESS(nextFreeBlock)));
578 
579  linkFreeBlocks(pgStart, prevFreeBlock, nextFreeBlock);
580 
581  // Check if we need to change the head of the blocks list
582  LD_SIZE(tempSz, ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, flIndex, SL_COUNT, slIndex));
583  FENCE_LOAD;
584  if(tempSz == freeBlock.value) {
585  ST_SIZE(ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, flIndex, SL_COUNT, slIndex),
586  nextFreeBlock.value);
587 
588  // If the new head is nullBlock then we clear the bitmaps to
589  // indicate that we have no more blocks available here
590  if(GET_ADDRESS(nextFreeBlock) == ADDR_OF(pool_t, pgStart, nullBlock)) {
591  LD_SIZE(tempSz, ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, flIndex));
592  FENCE_LOAD;
593  tempSz &= ~(1 << slIndex); // Clear me bit
594  ST_SIZE(ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, flIndex), tempSz);
595 
596  // Check if slAvailOrNot is now 0
597  if(tempSz == 0) {
598  LD_SIZE(tempSz, ADDR_OF(pool_t, pgStart, flAvailOrNot));
599  FENCE_LOAD;
600 
601  tempSz &= ~(1 << flIndex);
602  ST_SIZE(ADDR_OF(pool_t, pgStart, flAvailOrNot), tempSz);
603  }
604  }
605  FENCE_STORE;
606  }
607 }
608 
609 void addFreeBlock(u64 pgStart, headerAddr_t freeBlock) {
610  /* header_t* */ u64 freeBlockPtr = GET_ADDRESS(freeBlock);
611  int flIndex, slIndex;
612  tlsfSize_t tempSz;
613 
614  LD_SIZE(tempSz, ADDR_OF(header_t, freeBlockPtr, sizeBlock));
615  FENCE_LOAD;
616  mappingInsert(tempSz, &flIndex, &slIndex);
617 
618  headerAddr_t currentHead = { .address = 0ULL }, temp = { .address = 0ULL };
619 
620  LD_SIZE(currentHead.value, ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, flIndex, SL_COUNT, slIndex));
621  FENCE_LOAD;
622 
623  ASSERT(GET_ADDRESS(currentHead));
624  ASSERT(freeBlockPtr);
625  ASSERT(freeBlockPtr != ADDR_OF(pool_t, pgStart, nullBlock));
626 
627  // Set the links properly
628  SET_VALUE(temp, ADDR_OF(pool_t, pgStart, nullBlock));
629 
630  ST_SIZE(ADDR_OF(header_t, freeBlockPtr, prevFreeBlock), temp.value);
631  FENCE_STORE;
632  linkFreeBlocks(pgStart, freeBlock, currentHead);
633 
634  ST_SIZE(ADDR_OF(header_t, freeBlockPtr, prevFreeBlock), temp.value);
635  ST_SIZE(ADDR_OF(header_t, freeBlockPtr, nextFreeBlock), currentHead.value);
636  ST_SIZE(ADDR_OF(header_t, GET_ADDRESS(currentHead), prevFreeBlock), freeBlock.value); // Does not matter
637  // if the head was the
638  // nullBlock,
639  // that value is ignored anyways
640 
641  // Update the bitmaps
642  ST_SIZE(ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, flIndex, SL_COUNT, slIndex), freeBlock.value);
643 
644  LD_SIZE(tempSz, ADDR_OF(pool_t, pgStart, flAvailOrNot));
645  FENCE_LOAD;
646 
647  LD_SIZE(tempSz, ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, flIndex));
648  FENCE_LOAD;
649  if(!(tempSz & (1 << slIndex))) {
650  // Here it wasn't stored so we definitely have to update
651 
652  tempSz |= (1 << slIndex);
653  ST_SIZE(ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, flIndex), tempSz);
654 
655  LD_SIZE(tempSz, ADDR_OF(pool_t, pgStart, flAvailOrNot));
656  FENCE_LOAD;
657  if(!(tempSz & (1 << flIndex))) {
658  tempSz |= (1 << flIndex);
659  ST_SIZE(ADDR_OF(pool_t, pgStart, flAvailOrNot), tempSz);
660  }
661  }
662 
663  FENCE_STORE;
664 }

666 /* Split origBlock so that origBlock is realSize long and
667  * returns a pointer to the remaining block which is free
668  */
669 static headerAddr_t splitBlock(u64 pgStart, headerAddr_t origBlock, tlsfSize_t realSize) {
670  headerAddr_t remainingBlock = { .address = 0ULL };
671 
672  /* header_t* */ u64 origBlockPtr = GET_ADDRESS(origBlock);
673  tlsfSize_t origBlockSize;
674  LD_SIZE(origBlockSize, ADDR_OF(header_t, origBlockPtr, sizeBlock));
675  FENCE_LOAD;
676  ASSERT(origBlockSize > realSize + GminBlockRealSize);
677 
678  tlsfSize_t remainingSize = origBlockSize - realSize - GusedBlockOverhead;
679 
680  SET_VALUE(remainingBlock,
681  (origBlockPtr + GoffsetToStart + (realSize << ELEMENT_SIZE_LOG2)));
682 
683  // Take care of the remaining block (set its size and mark it as free)
684  ST_SIZE(ADDR_OF(header_t, GET_ADDRESS(remainingBlock), sizeBlock), remainingSize);
685  FENCE_STORE;
686  markBlockFree(pgStart, GET_ADDRESS(remainingBlock));
687 
688  // Set the size of the original block properly
689  ST_SIZE(ADDR_OF(header_t, origBlockPtr, sizeBlock), realSize);
690  FENCE_STORE;
691 
692  return remainingBlock;
693 }

695 /* Absorbs nextBlock into freeBlock making one much
696  * larger freeBlock. nextBlock must be physically next
697  * to freeBlock
698  */
699 static void absorbNext(u64 pgStart, headerAddr_t freeBlock, headerAddr_t nextBlock) {
700  /* header_t * */ u64 freeBlockPtr = GET_ADDRESS(freeBlock);
701  /* header_t * */ u64 nextBlockPtr = GET_ADDRESS(nextBlock);
702  tlsfSize_t tempSz, tempSz2;
703 
704  ASSERT(isBlockFree(freeBlockPtr));
705  ASSERT(isBlockFree(nextBlockPtr));
706  ASSERT(getNextBlock(pgStart, freeBlockPtr).value == nextBlock.value);
707 
708  LD_SIZE(tempSz, ADDR_OF(header_t, freeBlockPtr, sizeBlock));
709  LD_SIZE(tempSz2, ADDR_OF(header_t, nextBlockPtr, sizeBlock));
710  FENCE_LOAD;
711  tempSz += tempSz2 + GusedBlockOverhead; // freeBlockPtr->sizeBlock += nextBlockPtr->sizeBlock + GusedBlockOverhead
712  ST_SIZE(ADDR_OF(header_t, freeBlockPtr, sizeBlock), tempSz);
713  FENCE_STORE;
714  markBlockFree(pgStart, freeBlockPtr); // will update the other field
715 }

717 /* Merge with the previous block if it is free (toBeFreedBlock must
718  * be used so isPrevBlockFree can return true)
719  * Returns the resulting free block (either toBeFreedBlock marked as free
720  * or the larger block)
721  */
722 static headerAddr_t mergePrevious(u64 pgStart, headerAddr_t toBeFreedBlock) {
723  /* header_t * */ u64 toBeFreedBlockPtr = GET_ADDRESS(toBeFreedBlock);
724 
725  ASSERT(!isBlockFree(toBeFreedBlockPtr));
726  if(isPrevBlockFree(toBeFreedBlockPtr)) {
727  // Get the previous block
728  headerAddr_t prevBlock = getPrevBlock(pgStart, toBeFreedBlockPtr);
729  // Remove it from the free-lists (since it will be made bigger)
730  removeFreeBlock(pgStart, prevBlock);
731 
732  // Now we mark the toBeFreedBlock as free and merge it with the other one
733  markBlockFree(pgStart, toBeFreedBlockPtr);
734  absorbNext(pgStart, prevBlock, toBeFreedBlock);
735  toBeFreedBlock.value = prevBlock.value;
736  } else {
737  markBlockFree(pgStart, toBeFreedBlockPtr);
738  }
739 
740  ASSERT(isBlockFree(toBeFreedBlockPtr));
741  return toBeFreedBlock;
742 }

744 /* Merges a block with its block if that one is free as well. The input
745  * block must be free to start with
746  */
747 static headerAddr_t mergeNext(u64 pgStart, headerAddr_t freeBlock) {
748  ASSERT(isBlockFree(GET_ADDRESS(freeBlock)));
749  headerAddr_t nextBlock = getNextBlock(pgStart, GET_ADDRESS(freeBlock));
750  if(isBlockFree(GET_ADDRESS(nextBlock))) {
751  removeFreeBlock(pgStart, nextBlock);
752  absorbNext(pgStart, freeBlock, nextBlock);
753  }
754 
755  return freeBlock;
756 }

758 /* Pool construction */
759 static void initializePool(u64 pgStart, tlsfSize_t poolRealSize) {
760 
761  headerAddr_t mainBlockAddr, nullBlockAddr;
762 
763  SET_VALUE(nullBlockAddr, ADDR_OF(pool_t, pgStart, nullBlock));
764  SET_VALUE(mainBlockAddr, ADDR_OF(pool_t, pgStart, mainBlock));
765 
766  // Initialize the bitmaps to 0
767 
768  ST_SIZE(ADDR_OF(pool_t, pgStart, flAvailOrNot), 0);
769  int i, j;
770  for(i=0; i<FL_COUNT; ++i) {
771  ST_SIZE(ADDR_OF_1(pool_t, pgStart, tlsfSize_t, slAvailOrNot, i), 0);
772  for(j=0; j<SL_COUNT; ++j) {
773  ST_SIZE(ADDR_OF_2(pool_t, pgStart, tlsfSize_t, blocks, i, SL_COUNT, j), nullBlockAddr.value);
774  }
775  }
776 
777  // Initialize the main block properly
778  // We do something a little special in the main block: we take a little bit
779  // of it at the end to create a zero-sized block that is used
780  // to act as a sentinel. We therefore need GusedBlockOverhead space in the
781  // end (the 2* is because we also account for our own overhead)
782  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, mainBlock), sizeBlock),
783  poolRealSize - 2*GusedBlockOverhead);
784 
785  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, mainBlock), nextFreeBlock), nullBlockAddr.value);
786 
787  // This code is from markBlockFree (it marks mainBlock as free)
788  // We paste it here to remove some of the actions on next/prev block
789  // which bother valgrind
790  {
791  u64 locationToWrite = ADDR_OF(pool_t, pgStart, mainBlock) +
792  ((poolRealSize - GusedBlockOverhead) << ELEMENT_SIZE_LOG2) - sizeof(tlsfSize_t);
793  ST_SIZE(locationToWrite, poolRealSize - 2*GusedBlockOverhead);
794 
795  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, mainBlock), prevFreeBlock), 0xBEEF); // Some value that is not 0 or 1 for now
796  }
797  FENCE_STORE;
798 
799  // Add the sentinel
800  u64 sentinel = (ADDR_OF(pool_t, pgStart, mainBlock) +
801  ((poolRealSize - GusedBlockOverhead) << ELEMENT_SIZE_LOG2));
802 
803  ST_SIZE(ADDR_OF(header_t, sentinel, sizeBlock), 0);
804  ST_SIZE(ADDR_OF(header_t, sentinel, prevFreeBlock), 1); // This is mark prevBlockFree for sentinel
805 
806  // Initialize the nullBlock properly
807  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, nullBlock), sizeBlock), 0);
808  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, nullBlock), prevFreeBlock), nullBlockAddr.value);
809  ST_SIZE(ADDR_OF(header_t, ADDR_OF(pool_t, pgStart, nullBlock), nextFreeBlock), nullBlockAddr.value);
810  FENCE_STORE;
811 
812  // Add the big free block
813  addFreeBlock(pgStart, mainBlockAddr);
814 }

816 static u32 tlsfInit(u64 pgStart, u64 size) {
817  tlsfSize_t realSizeForPool = size >> ELEMENT_SIZE_LOG2;
818  /* The memory will be layed out as follows:
819  * - at location: the pool structure is used
820  * - the first free block starts right after that (aligned)
821  */
822 
823  tlsfSize_t poolHeaderSize = sizeof(pool_t) + ALIGN_BLOCK_BYTES - 1;
824  poolHeaderSize = (poolHeaderSize & ~(ALIGN_BLOCK_BYTES - 1));
825 
826  // Now we have a poolHeaderSize that is big enough to contain the pool
827  // and right after it, we can start the big block.
828 
829  tlsfSize_t poolRealSize = realSizeForPool -
830  ((poolHeaderSize + ELEMENT_SIZE_BYTES - 1) >> ELEMENT_SIZE_LOG2);
831 
832  if(poolRealSize < GminBlockRealSize || poolRealSize > GmaxBlockRealSize) {
833  DPRINTF(DEBUG_LVL_WARN, "Space mismatch allocating TLSF pool at 0x%"PRIx64" of sz %d (user sz: %d)\n",
834  pgStart, (u32)poolRealSize, (u32)(poolRealSize << ELEMENT_SIZE_LOG2));
835  DPRINTF(DEBUG_LVL_WARN, "Sz must be at least %d and at most %d\n", GminBlockRealSize, GmaxBlockRealSize);
836  return -1; // Can't allocate pool
837  }
838 
839  DPRINTF(DEBUG_LVL_INFO,"Allocating a TLSF pool at 0x%"PRIx64" of sz %d (user sz: %d)\n",
840  pgStart, (u32)poolRealSize, (u32)(poolRealSize << ELEMENT_SIZE_LOG2));
841 
842  initializePool(pgStart, poolRealSize);
843 
844  return 0;
845 }

852 static u64 tlsfMalloc(u64 pgStart, u64 size) {
853  u64 result = 0ULL;
854  tlsfSize_t allocSize, returnedSize;
855  int flIndex, slIndex;
856  headerAddr_t freeBlock, remainingBlock;
857 
858  allocSize = getRealSizeOfRequest(size);
859 
860  if(allocSize == 0 && size != 0) {
861  DPRINTF(DEBUG_LVL_VERB, "tlsfMalloc @0x%"PRIx64" returning NULL for too large size %"PRIu64"\n",
862  pgStart, size);
863  return _NULL;
864  }
865 
866  freeBlock = findFreeBlockForRealSize(pgStart, allocSize, &flIndex, &slIndex);
867  DPRINTF(DEBUG_LVL_VERB, "tslf_malloc @0x%"PRIx64" found a free block at 0x%"PRIx64"\n", pgStart,
868  addressForBlock(GET_ADDRESS(freeBlock)));
869  /* header_t * */ u64 freeBlockPtr = GET_ADDRESS(freeBlock);
870 
871  if(IS_NULL(freeBlockPtr)) {
872  DPRINTF(DEBUG_LVL_VERB, "tlsfMalloc @ 0x%"PRIx64" returning NULL for size %"PRIu64"\n",
873  pgStart, size);
874  return _NULL;
875  }
876  removeFreeBlock(pgStart, freeBlock);
877  LD_SIZE(returnedSize, ADDR_OF(header_t, freeBlockPtr, sizeBlock));
878  FENCE_LOAD;
879 
880  if(returnedSize > allocSize + GminBlockRealSize) {
881  remainingBlock = splitBlock(pgStart, freeBlock, allocSize);
882  DPRINTF(DEBUG_LVL_VERB, "tlsfMalloc @0x%"PRIx64" split block and re-added to free list 0x%"PRIx64"\n", pgStart,
883  addressForBlock(GET_ADDRESS(remainingBlock)));
884  addFreeBlock(pgStart, remainingBlock);
885  }
886 
887  markBlockUsed(pgStart, freeBlockPtr);
888 
889  result = addressForBlock(freeBlockPtr);
890  DPRINTF(DEBUG_LVL_VERB, "tlsfMalloc @ 0x%"PRIx64" returning 0x%"PRIx64" for size %"PRIu64"\n",
891  pgStart, result, size);
892  return result;
893 }

895 static void tlsfFree(u64 pgStart, u64 ptr) {
896  headerAddr_t bl;
897 
898  DPRINTF(DEBUG_LVL_VERB, "tlsfFree @ 0x%"PRIx64" going to free 0x%"PRIx64"\n",
899  pgStart, ptr);
900 
901  bl = blockForAddress(pgStart, ptr);
902 
903  bl = mergePrevious(pgStart, bl);
904  bl = mergeNext(pgStart, bl);
905  addFreeBlock(pgStart, bl);
906 }

908 static u64 tlsfRealloc(u64 pgStart, u64 ptr, u64 size) {
909  // First deal with corner cases:
910  // - non-zero size with null pointer is like malloc
911  // - zero size with ptr like free
912 
913  if(ptr && size == 0) {
914  tlsfFree(pgStart, ptr);
915  return _NULL;
916  }
917  if(!ptr) {
918  return tlsfMalloc(pgStart, size);
919  }
920 
921  u64 result;
922  result = _NULL;
923  headerAddr_t bl, nextBlock;
924  /* header_t * */ u64 blockAddr, nextBlockAddr;
925  tlsfSize_t realReqSize, realAvailSize, tempSz;
926 
927  bl = blockForAddress(pgStart, ptr);
928  blockAddr = GET_ADDRESS(bl);
929  nextBlock = getNextBlock(pgStart, blockAddr);
930  nextBlockAddr = GET_ADDRESS(nextBlock);
931 
932  LD_SIZE(realAvailSize, ADDR_OF(header_t, nextBlockAddr, sizeBlock));
933  LD_SIZE(tempSz, ADDR_OF(header_t, blockAddr, sizeBlock));
934  FENCE_LOAD;
935  realAvailSize += tempSz + GusedBlockOverhead; // realAvailSize = blockAddr->sizeBlock
936  // + nextBlockAddr->sizeBlock + GusedBlockOverhead
937  realReqSize = getRealSizeOfRequest(size);
938 
939  if(realReqSize > tempSz || (!isBlockFree(nextBlockAddr) || realReqSize > realAvailSize)) {
940  // We need to reallocate and copy
941  // Note, does not matter if pgStart was already truncated, it is an idempotent operation
942  result = tlsfMalloc(pgStart, size);
943  if(result) {
944  u64 sizeToCopy = tempSz<realReqSize?tempSz:realReqSize;
945  MALLOC_COPY((void*)result, (void*)ptr, sizeToCopy << ELEMENT_SIZE_LOG2);
946 
947  tlsfFree(pgStart, ptr); // Free the original block
948  }
949  } else {
950  if(realReqSize > tempSz) {
951  // This means we need to extend to the other block
952  removeFreeBlock(pgStart, nextBlock);
953  ST_SIZE(ADDR_OF(header_t, blockAddr, sizeBlock), realAvailSize);
954  FENCE_STORE;
955  markBlockUsed(pgStart, blockAddr);
956  } else {
957  realAvailSize = tempSz;
958  }
959  // We can trim to just the size used to create a new
960  // free block and reduce internal fragmentation
961  if(realAvailSize > realReqSize + GminBlockRealSize) {
962  headerAddr_t remainingBlock = splitBlock(pgStart, bl, realReqSize);
963  addFreeBlock(pgStart, remainingBlock);
964  }
965  result = ptr;
966  }
967 
968  return result;
969 }
