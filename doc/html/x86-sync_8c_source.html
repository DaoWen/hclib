144 static u64 pushTailX86(ocrQueue_t *self, u64 val) {
145  ocrQueueX86_t *rself = (ocrQueueX86_t*)self;
146  u8 success = 0;
147  s32 size;
148  while(!success) {
149  size = rself->tail - rself->head;
150  if(size == rself->size) {
151  ASSERT(0); // Deque full
152  return 0;
153  }
154  if(__sync_bool_compare_and_swap(&(rself->lock), 0, 1)) {
155  success = 1;
156  rself->content[rself->tail % rself->size] = val;
157  rself->tail = rself->tail + 1;
158  asm volatile("mfence");
159  rself->lock = 0;
160  }
161  }
162  return size + 1;
163 }
